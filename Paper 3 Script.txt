"""
Paper 3 Pipeline (BERT-first, reproducible, model-specific corpora)

What this script does
---------------------
- Loads configuration (config.yaml) and tokenisation filters (filters.yaml).
- Uses model-specific corpora: BERT reads `corpus_file_bert_cased[_cl]`,
  W2V reads `corpus_file_w2v[_cl]` for each mode in `paper3.corpora` (e.g., FULL, CLEAN).
- Builds tokenised sentences, lemmas, and raw sentences (SpaCy, deterministic).
- BERT: estimates, for each pair (A, B) and each concept set, whether A is closer
  to the concept centroid than B, via bootstrap on contextual embeddings.
  Two bootstrap engines:
    1) Fast "contexts" bootstrap with precomputed occurrence embeddings (preferred).
    2) Fallback sentence bootstrap (slower), if precompute is not enabled.
- W2V (optional): provides baseline differences (or retrain bootstrap if requested).
- Applies multiple-testing corrections (BH-FDR, BY-FDR, Holm) to bootstrap p-values.
- Writes:
    * results_paper3/<run_tag>/bootstrap_results_with_fdr.csv
    * results_paper3/<run_tag>/diagnostics_*.json
    * results_paper3/<run_tag>/figures/bootstrap_*_hist.png and .svg
    * results_paper3/<run_tag>/run_manifest_paper3.json (reproducibility metadata)

Determinism
-----------
- Single-thread BLAS/OMP is forced (OPENBLAS, MKL, OMP, NUMEXPR = 1).
- PYTHONHASHSEED is fixed.
- PyTorch deterministic algorithms are enabled.
- TF32 is disabled.
- cuDNN benchmarking is disabled.

Config keys used
----------------
See `config.yaml`:
paper3:
  bootstrap_mode: "fast"|"slow"
  fast_bootstrap_iterations: int
  slow_bootstrap_iterations: int
  corpora: ["FULL", "CLEAN"]         # which modes to run
  models: ["bert", "cbow", "skipgram"]
  w2v:
    fast: {...}                       # params for "fast" mode
    slow: {...}                       # params for "slow" mode
  bert:
    max_contexts_per_word_fast: int
    max_contexts_per_word_slow: int
    use_last_hidden_layer: bool
    fp16_inference: bool
    # advisory (no-op) provenance flags kept for parity across papers:
    use_final: bool
    source: str
    prefer_best_epoch: bool
    pooling: str
    layer: str
  w2v_bootstrap_mode: "off"|"baseline"|"retrain"
  use_validated_cbow_baseline: bool
  permutation_iterations: int
  # precompute (fast contexts bootstrap)
  bootstrap_unit: "contexts"|"sentences"
  precompute_cache_dir: str
  precompute_batch_size: int

Top-level:
  seed: int
  corpus_dir: str
  corpus_file_w2v: str
  corpus_file_w2v_cl: str
  corpus_file_bert_cased: str
  corpus_file_bert_cased_cl: str
  bootstrap_test_pairs: [[str,str], ...]
  bootstrap_concepts_to_test: {name: [words...]}

Run:
  python Paper3_final.py --config config.yaml --filters filters.yaml
"""

def _make_logger(log_dir: Path) -> logging.Logger:
    """Create a console+file logger for the pipeline.

    Parameters
    ----------
    log_dir : Path
        Directory where the log file will be written.

    Returns
    -------
    logging.Logger
        Configured logger named 'paper3'.
    """
    logger = logging.getLogger("paper3")
    logger.setLevel(logging.INFO)
    if logger.handlers:
        return logger
    fmt = logging.Formatter("[%(asctime)s] %(levelname)s | %(message)s", "%H:%M:%S")
    ch = logging.StreamHandler(sys.stdout); ch.setFormatter(fmt); ch.setLevel(logging.INFO)
    fh = logging.FileHandler(log_dir / "paper3.log", encoding="utf-8"); fh.setFormatter(fmt); fh.setLevel(logging.INFO)
    logger.addHandler(ch); logger.addHandler(fh)
    return logger

LOGGER = _make_logger(LOGDIR)

def rebase_outputs_to_run_tag(run_tag: str) -> None:
    """Repoint RES/FIG/LOGDIR/MASTER into results_paper3/<run_tag>/..., and retarget the file logger."""
    global RES, FIG, LOGDIR, MASTER, LOGGER
    base_res = Path(args.outres)
    RES = base_res / run_tag
    FIG = RES / "figures"
    LOGDIR = RES / "logs"
    MASTER = RES / "master"
    for d in (RES, FIG, LOGDIR, MASTER):
        d.mkdir(parents=True, exist_ok=True)
    # replace file handler so logs go under this run folder
    for h in list(LOGGER.handlers):
        if isinstance(h, logging.FileHandler):
            LOGGER.removeHandler(h)
            try:
                h.close()
            except (OSError, ValueError):
                pass

    fmt = logging.Formatter("[%(asctime)s] %(levelname)s | %(message)s", "%H:%M:%S")
    fh = logging.FileHandler(LOGDIR / "paper3.log", encoding="utf-8")
    fh.setFormatter(fmt); fh.setLevel(logging.INFO)
    LOGGER.addHandler(fh)

def _read_yaml(path: Path) -> Dict[str, Any]:
    """Read a YAML file into a dictionary.

    Raises
    ------
    FileNotFoundError
        If the path does not exist.

    Returns
    -------
    dict
        Parsed YAML mapping, empty dict if file is empty.
    """
    if not path.is_file():
        raise FileNotFoundError(f"Missing YAML: {path}")
    with io.open(path, encoding="utf-8") as f:
        return yaml.safe_load(f) or {}

def _parse_paper3_config(raw: Dict[str, Any]) -> Paper3Config:
    """Construct Paper3Config from raw YAML.

    Applies CLI overrides for speed tier, permutation iterations, and model toggles.
    """
    p3 = raw.get("paper3", {}) or {}
    w2v_fast = W2VParams(**(p3.get("w2v", {}).get("fast", {}) or {}))
    w2v_slow = W2VParams(**(p3.get("w2v", {}).get("slow", {}) or {}))
    bert = BertParams(**(p3.get("bert", {}) or {}))
    require = RequireVersions(**(p3.get("require_versions", {}) or {}))

    perm_cfg = int(p3.get("permutation_iterations", 10000))
    if args.perm_iters is not None:
        perm_cfg = int(args.perm_iters)

    cli_mode = "fast" if args.fast else ("slow" if args.slow else str(p3.get("bootstrap_mode", "fast")).lower())

    cfg = Paper3Config(
        bootstrap_mode=cli_mode,
        fast_bootstrap_iterations=int(p3.get("fast_bootstrap_iterations", 200)),
        slow_bootstrap_iterations=int(p3.get("slow_bootstrap_iterations", 1000)),
        corpora=[str(m).upper() for m in (p3.get("corpora", ["FULL"]) or [])],
        models=[m.lower() for m in (p3.get("models", ["bert"]) or [])],
        w2v_fast=w2v_fast,
        w2v_slow=w2v_slow,
        bert=bert,
        w2v_bootstrap_mode=str(p3.get("w2v_bootstrap_mode", "off")).lower(),
        use_validated_cbow_baseline=bool(p3.get("use_validated_cbow_baseline", True)),
        permutation_iterations=perm_cfg,
        require_versions=require,
        bootstrap_unit=str(p3.get("bootstrap_unit", "contexts")).lower(),
        target_context_sampling=str(p3.get("target_context_sampling", "observed")).lower(),
        use_fdr_for_publication_status=bool(p3.get("use_fdr_for_publication_status", True)),
        precompute_cache_dir=str(p3.get("precompute_cache_dir", "cache_paper3_bert")),
        precompute_batch_size=int(p3.get("precompute_batch_size", 64)),
    )

    # Apply CLI model toggles
    if args.bert_flag and "bert" not in cfg.models:
        cfg.models.append("bert")
    if not args.bert_flag and "bert" in cfg.models:
        cfg.models.remove("bert")
    if not args.w2v_flag:
        cfg.models = [m for m in cfg.models if m not in ("cbow", "skipgram", "w2v")]
        cfg.w2v_bootstrap_mode = "off"
    if args.w2v_mode is not None:
        cfg.w2v_bootstrap_mode = args.w2v_mode

    return cfg

# Pairs & concepts
def _parse_bootstrap_groups(cfg: Dict[str, Any]) -> Dict[str, Dict[str, Any]]:
    """
    Support either:
      1) new grouped config under `bootstrap_groups`
      2) legacy top-level `bootstrap_test_pairs` + `bootstrap_concepts_to_test`
    """
    groups_raw = cfg.get("bootstrap_groups", {}) or {}
    if groups_raw:
        out: Dict[str, Dict[str, Any]] = {}
        for group_name, group in groups_raw.items():
            pairs = [(str(a), str(b)) for a, b in (group.get("pairs", []) or [])]
            concepts = {str(k): [str(x) for x in (v or [])] for k, v in (group.get("concepts", {}) or {}).items()}
            out[str(group_name)] = {
                "pairs": pairs,
                "concepts": concepts,
            }
        return out

    legacy_pairs = [(str(a), str(b)) for a, b in (cfg.get("bootstrap_test_pairs", []) or [])]
    legacy_concepts = {str(k): [str(x) for x in (v or [])] for k, v in (cfg.get("bootstrap_concepts_to_test", {}) or {}).items()}
    return {
        "default": {
            "pairs": legacy_pairs,
            "concepts": legacy_concepts,
        }
    }

def _concept_guard_key(value: Any) -> str:
    """Normalize a seed term for direct pair-term leakage checks."""
    text = str(value).strip().lower()
    for k, v in NORMALISATION_MAP.items():
        if k and v:
            text = text.replace(str(k).strip().lower(), str(v).strip().lower())
    text = text.replace("ō", "o").replace("ū", "u")
    text = re.sub(r"[^a-z0-9]+", "", text)
    return text


def _remove_pair_term_leakage_from_groups(
    groups: Dict[str, Dict[str, Any]]
) -> Dict[str, Dict[str, Any]]:
    """
    Remove direct pair terms from concept seed lists.

    This prevents circular tests such as comparing engawa/veranda against a
    centroid that contains "veranda".
    """
    cleaned_groups: Dict[str, Dict[str, Any]] = {}

    for group_name, group in groups.items():
        pairs = [(str(a), str(b)) for a, b in group.get("pairs", [])]
        concepts = group.get("concepts", {}) or {}

        banned: Set[str] = set()
        for term1, term2 in pairs:
            banned.add(_concept_guard_key(term1))
            banned.add(_concept_guard_key(term2))

        cleaned_concepts: Dict[str, List[str]] = {}

        for concept_name, words in concepts.items():
            kept: List[str] = []
            removed: List[str] = []

            for word in words:
                if _concept_guard_key(word) in banned:
                    removed.append(str(word))
                else:
                    kept.append(str(word))

            if removed:
                LOGGER.warning(
                    "[%s] Removed direct pair-term leakage from %s: %s",
                    group_name,
                    concept_name,
                    removed,
                )

            cleaned_concepts[str(concept_name)] = kept

        cleaned_groups[str(group_name)] = {
            "pairs": pairs,
            "concepts": cleaned_concepts,
        }

    return cleaned_groups

BOOTSTRAP_GROUPS = _remove_pair_term_leakage_from_groups(_parse_bootstrap_groups(CFG))

        N_BOOT,
    )

def sha256_file(path: Path) -> str:
    """Return SHA256 of a file, or empty string if missing."""
    try:
        h = hashlib.sha256()
        with path.open("rb") as fh:
            for chunk in iter(lambda: fh.read(1 << 20), b""):
                h.update(chunk)
        return h.hexdigest()
    except FileNotFoundError:
        return ""

def bert_weight_file(model_dir: Path) -> Path:
    """Return the BERT weight file used by a local checkpoint."""
    safetensors_path = model_dir / "model.safetensors"
    pytorch_path = model_dir / "pytorch_model.bin"

    if safetensors_path.is_file():
        return safetensors_path
    if pytorch_path.is_file():

    raise FileNotFoundError(f"No BERT weight file found in {model_dir}")

def check_required_versions(req: RequireVersions) -> None:
    """Optionally enforce minimum versions for core libraries.

    Raises
    ------
    RuntimeError

    """
    versions = {
        "numpy": Version(np.__version__),
        "torch": Version(torch.__version__),
        "spacy": Version(spacy.__version__),
    }
    try:
        import transformers  # type: ignore
        versions["transformers"] = Version(getattr(transformers, "__version__", "0.0.0"))
    except ImportError:
        pass
    for lib, need in req.__dict__.items():
        if not need:
            continue
        if lib not in versions:

            continue

class Timer:
    """Context manager for timing a named block."""
    def __init__(self, label: str):
        """Create a Timer with a label."""
        self.label = label
        self.t0 = time.perf_counter()
    def stop(self) -> float:
        """Stop the timer, log duration, and return seconds."""
        dt = time.perf_counter() - self.t0
        LOGGER.info("%s completed in %.3fs", self.label, dt)
        return dt
    def __enter__(self):
        """Enter the context and return self."""
        return self
    def __exit__(self, exc_type, exc, tb):
        """Exit the context and log the duration."""
        self.stop()

def cosine(a: NDArray[np.float_], b: NDArray[np.float_]) -> float:
    """Cosine similarity between two vectors."""
    na = float(np.linalg.norm(a)); nb = float(np.linalg.norm(b))
    if na == 0.0 or nb == 0.0:
        return 0.0
    return float(np.dot(a, b) / (na * nb))

def contains_japanese(text: str) -> bool:
    """Return True if the text contains Japanese script characters."""
    return bool(re_rx.search(r"[\p{Hiragana}\p{Katakana}\p{Han}]", text))

def preprocess_text(text: str) -> List[str]:
    """Apply normalisation map, whitelist, lowercase, and simple token merges."""
    t = text
    for k, v in NORMALISATION_MAP.items():
        t = t.replace(k, v)
    toks = CHAR_WHITELIST_RE.sub(" ", t.lower()).split()
    if not MERGE_RE_FOLLOWING:
        return toks
    merged: List[str] = []
    i = 0
    while i < len(toks):
        if toks[i] == "re" and i + 1 < len(toks):
            merged.append(toks[i] + toks[i + 1]); i += 2
        elif toks[i] != "re":
            merged.append(toks[i]); i += 1
        else:
            i += 1
    return merged

def build_tokenized_and_raw(corpus_path: Path, nlp: spacy.language.Language) -> Tuple[List[List[str]], List[str], List[List[str]]]:
    """Tokenise corpus into tokens and lemmas, and return aligned raw sentences.

    Returns
    -------
    (tokenized_sentences, raw_sentences, lemma_sentences)

    Notes
    -----
    The returned raw sentences are filtered to the same subset as the tokenised
    and lemmatised sentences, so index-based lookups stay aligned.
    """
    text = corpus_path.read_text(encoding="utf-8")
    all_raw_sents = [s.strip() for s in SENT_SPLIT_RE.split(text) if s.strip()]
    normed_for_tok = [" ".join(preprocess_text(s)) for s in all_raw_sents]

    tokenized_sentences: List[List[str]] = []
    raw_sents: List[str] = []
    lemma_sentences: List[List[str]] = []

    doc_iter = nlp.pipe(normed_for_tok)
    for raw_sent, doc in tqdm(
        zip(all_raw_sents, doc_iter),
        total=len(normed_for_tok),
        desc="Processing Corpus",
        leave=False,
    ):
        keep: List[str] = []
        lemmas: List[str] = []
        for tok in doc:
            if (tok.pos_ in ALLOWED_POS or contains_japanese(tok.text)) and not tok.is_stop:
                keep.append(tok.text.lower())
                lemmas.append((tok.lemma_ or tok.text).lower())

        if keep:
            tokenized_sentences.append(keep)
            raw_sents.append(raw_sent)
            lemma_sentences.append(lemmas)

    return tokenized_sentences, raw_sents, lemma_sentences

def cache_root_for(mode_tag: str) -> Path:
    """Return the cache root for a given mode tag (e.g., 'FULL')."""

def load_validated_w2v(mode_tag: str, kind: str) -> Optional[Word2VecType]:
    """Load a validated W2V model if available. Returns None if missing or unavailable."""
    if not W2V_AVAILABLE:
        LOGGER.warning("gensim not available, skipping W2V load")
        return None
    fname = "w2v_cbow.model" if kind == "cbow" else "w2v_sg.model"
    path = cache_root_for(mode_tag) / fname
    if path.exists():
        LOGGER.info("Loaded validated %s: %s (hash=%s)", kind.upper(), path, sha256_file(path))
        return cast(Word2VecType, GensimWord2Vec.load(str(path)))
    LOGGER.warning("Validated %s not found at %s", kind.upper(), path)
    return None

def train_temp_w2v(sentences: List[List[str]], sg_flag: int, params: W2VParams) -> Word2VecType:
    """Train and return a temporary W2V model. Raises if gensim unavailable."""
    if not W2V_AVAILABLE or GensimWord2Vec is None:
        raise RuntimeError("gensim Word2Vec is not installed but W2V was requested.")
    return GensimWord2Vec(
        sentences=sentences,
        vector_size=int(params.vector_size),
        window=int(params.window),
        min_count=int(params.min_count),
        workers=int(params.workers),
        sg=int(sg_flag),
        epochs=int(params.epochs),
        seed=SEED
    )

def w2v_alignment(target: str, concept_words: List[str], model: Word2VecType) -> float:
    """Cosine(target, centroid(concepts)) using a trained W2V model."""
    if not hasattr(model, "wv"):
        return 0.0
    if target not in model.wv:
        return 0.0
    concept_vecs = [model.wv[w] for w in concept_words if w in model.wv]
    if not concept_vecs:
        return 0.0
    centroid = np.mean(np.stack(concept_vecs), axis=0)
    return cosine(model.wv[target], centroid)

def w2v_bootstrap_diff_retrain(word1: str, word2: str, corpus_sents: List[List[str]],
                               concept_words: List[str], n_boot: int,
                               sg_flag: int, params: W2VParams) -> Tuple[float, float, float, List[float]]:
    """Bootstrap alignment difference with per-sample re-training."""
    model_full = train_temp_w2v(corpus_sents, sg_flag, params)
    observed = w2v_alignment(word1, concept_words, model_full) - w2v_alignment(word2, concept_words, model_full)
    diffs: List[float] = []
    for _ in tqdm(range(n_boot), desc=f"W2V(sg={sg_flag}) {word1} vs {word2}", leave=False):
        sample = resample(corpus_sents, random_state=int(rng.integers(0, 2**31 - 1)))
        m = train_temp_w2v(sample, sg_flag, params)
        d = w2v_alignment(word1, concept_words, m) - w2v_alignment(word2, concept_words, m)
        diffs.append(d)
    if not diffs:
        return observed, 0.0, 0.0, []
    lo, hi = float(np.percentile(diffs, 2.5)), float(np.percentile(diffs, 97.5))
    return observed, lo, hi, diffs

def w2v_baseline_diff_loaded(model: Optional[Word2VecType], word1: str, word2: str, concept_words: List[str]) -> Optional[float]:
    """Compute a non-bootstrap baseline difference using a loaded W2V model."""
    if model is None:
        return None
    return w2v_alignment(word1, concept_words, model) - w2v_alignment(word2, concept_words, model)

def load_best_epoch_bert(mode_tag: str) -> Tuple[BertForMaskedLM, BertTokenizerFast, Path]:
    """Load the fine-tuned BERT for this mode.

    By default this fails hard if no fine-tuned checkpoint can be resolved.
    Set paper3.bert.allow_fallback_model: true only when a Hugging Face fallback
    is explicitly intended.
    """

    def _has_model_files(p: Path) -> bool:
        return (p / "config.json").is_file() and (
            (p / "pytorch_model.bin").is_file() or (p / "model.safetensors").is_file()
        )

    def _load_local(p: Path, label: str) -> Tuple[BertForMaskedLM, BertTokenizerFast, Path]:
        tok = BertTokenizerFast.from_pretrained(str(p), local_files_only=True)
        try:
            mdl = BertForMaskedLM.from_pretrained(
                str(p),
                attn_implementation="eager",
                local_files_only=True,
            )
        except TypeError:
            mdl = BertForMaskedLM.from_pretrained(str(p), local_files_only=True)
        mdl = cast(BertForMaskedLM, mdl).to(DEVICE)
        mdl.eval()
        LOGGER.info("Loaded fine-tuned BERT from %s: %s", label, p)
        return mdl, tok, p

    def _load_hf_fallback(reason: str) -> Tuple[BertForMaskedLM, BertTokenizerFast, Path]:
        if not bool(P3.bert.allow_fallback_model):
            raise RuntimeError(
                f"{reason}. No fine-tuned BERT checkpoint was resolved under {root}. "
                "Set paper3.bert.allow_fallback_model: true only if you explicitly want "
                "to run with the Hugging Face fallback."
            )

        fallback_id = P3.bert.fallback_model_id or "bert-base-cased"
        LOGGER.warning("%s. Using Hugging Face fallback: %s", reason, fallback_id)

        tok = BertTokenizerFast.from_pretrained(fallback_id)
        try:
            mdl = BertForMaskedLM.from_pretrained(fallback_id, attn_implementation="eager")
        except TypeError:
            mdl = BertForMaskedLM.from_pretrained(fallback_id)

        mdl = cast(BertForMaskedLM, mdl).to(DEVICE)
        mdl.eval()

        fallback_dir = root / "fallback_hf"
        fallback_dir.mkdir(parents=True, exist_ok=True)
        tok.save_pretrained(str(fallback_dir))
        mdl.save_pretrained(str(fallback_dir))
        LOGGER.info("Saved fallback model to %s", fallback_dir)
        return mdl, tok, fallback_dir

    root = cache_root_for(mode_tag) / "bert_cased"

    if not root.is_dir():
        return _load_hf_fallback(f"Model root missing: {root}")

    # 1) trainer_state.json -> best checkpoint
    state_path = root / "trainer_state.json"
    if state_path.is_file():
        try:
            state = json.loads(state_path.read_text(encoding="utf-8"))
            best_rel = state.get("best_model_checkpoint", "").strip()
        except json.JSONDecodeError as e:
            raise RuntimeError(f"Cannot parse {state_path}: {e}") from e

        if best_rel:
            best_path = Path(best_rel)
            ckpt = best_path if best_path.is_absolute() else (root / best_rel)
            if ckpt.is_dir() and _has_model_files(ckpt):
                return _load_local(ckpt, "trainer_state.json")
            raise FileNotFoundError(
                f"best_model_checkpoint in {state_path} points to a missing or invalid directory: {ckpt}"
            )

    # 2) single unambiguous checkpoint-* directory
    ckpts = [p for p in root.glob("checkpoint-*") if p.is_dir() and _has_model_files(p)]

    final_dir = root / "final"
    final_ok = final_dir.is_dir() and _has_model_files(final_dir)

    if len(ckpts) == 1:
        return _load_local(ckpts[0], "sole checkpoint dir")

    if len(ckpts) > 1 and not final_ok:
        raise RuntimeError(
            f"Multiple checkpoint directories were found under {root}, but trainer_state.json "
            "did not resolve a unique best checkpoint and no valid final/ directory exists."
        )

    if final_ok:
        return _load_local(final_dir, "'final' directory")

    return _load_hf_fallback(f"No fine-tuned checkpoint under {root}")

_BERT_EMBED_CACHE: Dict[Tuple[str, str, bool], np.ndarray] = {}

def _normalise_match_key(text: str) -> str:
    """Lowercase and apply configured normalisation map for matching only."""
    out = text.lower()
    for k, v in NORMALISATION_MAP.items():
        if k and v:
            out = out.replace(str(k).lower(), str(v).lower())
    return out

def _surface_variants_for_term(term: str) -> List[str]:
    """Return surface variants that should count as the same target for matching."""
    t_low = term.lower()
    target_key = _normalise_match_key(t_low)
    variants: Set[str] = {t_low, target_key}
    for k, v in NORMALISATION_MAP.items():
        k_low = str(k).lower()
        v_low = str(v).lower()
        if _normalise_match_key(k_low) == target_key or _normalise_match_key(v_low) == target_key:
            variants.add(k_low)
            variants.add(v_low)
    return sorted({v for v in variants if v}, key=len, reverse=True)

def _tolerant_term_pattern(term: str) -> re.Pattern:
    """Allow hyphen, space, and underscore variation while preserving word boundaries."""
    t = re.escape(term.lower())
    t = t.replace(r"\-", r"[-_\s]?").replace(r"\ ", r"[-_\s]?").replace(r"_", r"[-_\s]?")
    return re.compile(rf"(?<!\w){t}(?!\w)", re.IGNORECASE)

def _find_term_spans(sentence: str, term: str) -> List[Tuple[int, int]]:
    """Find target spans in the raw sentence using tolerant matching and configured variants."""
    sent_low = sentence.lower()
    spans: List[Tuple[int, int]] = []
    seen: Set[Tuple[int, int]] = set()
    for variant in _surface_variants_for_term(term):
        pat = _tolerant_term_pattern(variant)
        for m in pat.finditer(sent_low):
            sp = (int(m.start()), int(m.end()))
            if sp not in seen:
                seen.add(sp)
                spans.append(sp)
    spans.sort()
    return spans

def collect_context_indices(raw_sents: List[str], lemma_sents: List[List[str]], target_word: str, max_k: int) -> List[int]:
    """Pick up to max_k sentence indices that contain the target by lemma or tolerant surface match."""
    t_low = target_word.lower()
    idxs = [j for j, lemmas in enumerate(lemma_sents) if t_low in lemmas]
    if not idxs:
        idxs = [j for j, s in enumerate(raw_sents) if _find_term_spans(s, t_low)]
    if not idxs:
        return []
    if len(idxs) > max_k:
        idxs = list(rng.choice(idxs, size=max_k, replace=False))
    return idxs

@torch.no_grad()
def bert_sentence_token_embeddings(
    sentence: str,
    target_lower: str,
    model: BertForMaskedLM,
    tokenizer: BertTokenizerFast,
    use_last: bool = True
) -> Optional[np.ndarray]:
    """Return mean token embeddings for occurrences of the target within a sentence."""
    key = (sentence, target_lower, bool(use_last))
    if key in _BERT_EMBED_CACHE:
        return _BERT_EMBED_CACHE[key]

    enc = tokenizer(sentence, return_tensors="pt", return_offsets_mapping=True)
    for k in ("input_ids", "attention_mask", "token_type_ids"):
        if k in enc:
            enc[k] = enc[k].to(DEVICE)
    offsets = enc.pop("offset_mapping")[0].tolist()

    if P3.bert.fp16_inference and DEVICE.type == "cuda":
        with torch.autocast(device_type="cuda", dtype=torch.float16):
            out = model(**enc, output_hidden_states=True)
    else:
        out = model(**enc, output_hidden_states=True)

    hidden = out.hidden_states[-1][0] if use_last else torch.stack(out.hidden_states[-4:]).mean(dim=0)[0]

    match_spans = _find_term_spans(sentence, target_lower)
    if not match_spans:
        return None

    def overlaps(x: Tuple[int, int], y: Tuple[int, int]) -> bool:
        """Return True if two half-open intervals overlap."""
        return x[0] < y[1] and y[0] < x[1]

    token_vecs: List[np.ndarray] = []
    for idx, (tok_start, tok_end) in enumerate(offsets):
        if tok_start == 0 and tok_end == 0:
            continue
        if any(overlaps((tok_start, tok_end), span) for span in match_spans):
            token_vecs.append(hidden[idx].detach().cpu().numpy())

    if not token_vecs:
        return None
    vec = np.mean(np.stack(token_vecs), axis=0)
    _BERT_EMBED_CACHE[key] = vec
    return vec

def _cache_key_for_precompute(words: Iterable[str], bert_src: Path, max_ctx: int, use_last: bool, corpus_hash: str) -> str:
    """Build a stable key for the precomputed embeddings NPZ file."""
    payload = {
        "words": sorted(set(w.lower() for w in words)),
        "bert_src": str(bert_src),
        "max_ctx": int(max_ctx),
        "use_last": bool(use_last),
        "corpus_hash": corpus_hash,
        "filters": asdict(FLT),
        "fp16_inference": bool(P3.bert.fp16_inference),
        "seed": int(SEED),
        "match_schema": "v3_tolerant_span_matching_equal_weight_concepts_seeded_cache",
    }
    blob = json.dumps(payload, sort_keys=True).encode("utf-8")
    return hashlib.sha256(blob).hexdigest()

def _save_precomputed(path: Path, data: Dict[str, np.ndarray]) -> None:
    """Save precomputed occurrence embeddings to a compressed NPZ."""
    path.parent.mkdir(parents=True, exist_ok=True)
    np.savez_compressed(str(path), **{k: v.astype(np.float32) for k, v in data.items()})

def _load_precomputed(path: Path) -> Optional[Dict[str, np.ndarray]]:
    """Load precomputed occurrence embeddings from NPZ if present."""
    if not path.is_file():
        return None
    with np.load(str(path), allow_pickle=False) as z:
        return {k: z[k] for k in z.files}

@torch.no_grad()
def precompute_bert_occurrence_embeddings(
    words: List[str],
    raw_sents: List[str],
    lemma_sents: List[List[str]],
    model: BertForMaskedLM,
    tokenizer: BertTokenizerFast,
    max_contexts: int,
    use_last: bool,
    batch_size: int = 64,
    fp16: bool = False,
) -> Dict[str, np.ndarray]:
    """Precompute per-occurrence contextual embeddings for a set of target words.

    Notes
    -----
    This accelerates the "contexts" bootstrap by avoiding repeated tokenisation
    and forward passes for the same sentences when multiple targets share them.
    """
    from collections import defaultdict

    t_low_all = [w.lower() for w in words]
    sent_to_targets: Dict[int, Set[str]] = defaultdict(set)

    for w in t_low_all:
        for sent_idx in collect_context_indices(raw_sents, lemma_sents, w, max_contexts):
            sent_to_targets[sent_idx].add(w)

    if not sent_to_targets:
        return {w: np.empty((0, 768), dtype=np.float32) for w in t_low_all}

    all_idxs = sorted(sent_to_targets.keys())
    out_lists: Dict[str, List[np.ndarray]] = {w: [] for w in t_low_all}

    i0 = 0
    while i0 < len(all_idxs):
        batch_idxs = all_idxs[i0:i0 + batch_size]
        batch_sents = [raw_sents[i] for i in batch_idxs]
        enc = tokenizer(batch_sents, return_tensors="pt", return_offsets_mapping=True, padding=True, truncation=True)
        offsets = enc.pop("offset_mapping")
        for k in ("input_ids", "attention_mask", "token_type_ids"):
            if k in enc:
                enc[k] = enc[k].to(DEVICE)

        with torch.autocast("cuda", enabled=(fp16 and DEVICE.type == "cuda")):
            out = model(**enc, output_hidden_states=True)
            hidden = out.hidden_states[-1] if use_last else torch.stack(out.hidden_states[-4:]).mean(dim=0)

        hidden = hidden.detach().cpu()

        for row_idx, sent_idx in enumerate(batch_idxs):
            sentence = batch_sents[row_idx]
            tok_spans = offsets[row_idx].tolist()
            token_matrix = hidden[row_idx]
            for w in sent_to_targets[sent_idx]:
                spans = _find_term_spans(sentence, w)
                if not spans:
                    continue
                pieces: List[np.ndarray] = []
                for tok_i, (a, b_) in enumerate(tok_spans):
                    if a == 0 and b_ == 0:
                        continue
                    for sa, sb in spans:
                        if a < sb and sa < b_:
                            pieces.append(token_matrix[tok_i].numpy())
                            break
                if pieces:
                    out_lists[w].append(np.mean(np.stack(pieces, axis=0), axis=0))

        i0 += batch_size

    any_key = next((k for k, v in out_lists.items() if v), None)
    dim = out_lists[any_key][0].shape[-1] if any_key is not None else 768
    return {
        w: (np.stack(v, axis=0).astype(np.float32) if len(v) else np.empty((0, dim), dtype=np.float32))
        for w, v in out_lists.items()
    }

def _mean_sample(x: np.ndarray, k: int) -> np.ndarray:
    """Sample k rows with replacement and return their mean; safe on empty input."""
    if x.size == 0:
        return np.zeros((x.shape[1] if x.ndim == 2 else 768,), dtype=np.float32)
    idx = rng.integers(0, x.shape[0], size=max(1, k))
    return x[idx].mean(axis=0)

def bert_bootstrap_diff_precomputed(
    word1: str,
    word2: str,
    concept_words: List[str],
    n_boot: int,
    pre: Dict[str, np.ndarray],
    target_context_sampling: str = "observed",
) -> Tuple[float, float, float, List[float]]:
    """
    Bootstrap using precomputed occurrence embeddings, with equal weighting across concept words.

    target_context_sampling:
      - "observed": resample each target using its observed context count.
      - "balanced": resample both target terms using min(n_word1, n_word2).
    """
    sampling = str(target_context_sampling or "observed").lower()
    if sampling not in {"observed", "balanced"}:
        raise ValueError("target_context_sampling must be 'observed' or 'balanced'")

    w1 = pre.get(word1.lower(), np.empty((0, 768), dtype=np.float32))
    w2 = pre.get(word2.lower(), np.empty((0, 768), dtype=np.float32))
    pool = [pre.get(w.lower(), np.empty((0, 768), dtype=np.float32)) for w in concept_words]

    dim = 768
    for arr in [w1, w2] + pool:
        if arr.ndim == 2 and arr.size:
            dim = arr.shape[1]
            break

    w1_m = w1.mean(axis=0) if w1.size else np.zeros((dim,), dtype=np.float32)
    w2_m = w2.mean(axis=0) if w2.size else np.zeros((dim,), dtype=np.float32)
    c_m = _concept_centroid_from_pool(pool)
    observed = cosine(w1_m, c_m) - cosine(w2_m, c_m)

    n1 = int(w1.shape[0]) if w1.ndim == 2 else 0
    n2 = int(w2.shape[0]) if w2.ndim == 2 else 0

    if sampling == "balanced" and n1 > 0 and n2 > 0:
        k1 = min(n1, n2)
        k2 = min(n1, n2)
    else:
        k1 = n1 if n1 > 0 else 1
        k2 = n2 if n2 > 0 else 1

    diffs: List[float] = []

    for _ in range(n_boot):
        c_b = _bootstrap_concept_centroid_from_pool(pool)
        w1_b = _mean_sample(w1, k1)
        w2_b = _mean_sample(w2, k2)
        diffs.append(cosine(w1_b, c_b) - cosine(w2_b, c_b))

    if not diffs:
        return observed, 0.0, 0.0, []

    lo, hi = np.percentile(np.asarray(diffs, dtype=float), [2.5, 97.5]).tolist()
    return observed, float(lo), float(hi), diffs

def label_permutation_p_precomputed(
    w1_mat: np.ndarray,
    w2_mat: np.ndarray,
    concept_centroid: np.ndarray,
    observed: float,
    n_perm: int,
    seed: int,
    target_context_sampling: str = "observed",
) -> float:
    """
    Two-sided label-permutation test for the difference in concept alignment.

    For target_context_sampling="balanced", each permutation compares two
    equal-sized pseudo-groups of size min(n_word1, n_word2).
    """
    if n_perm <= 0:
        return float("nan")

    if w1_mat.ndim != 2 or w2_mat.ndim != 2:
        return float("nan")

    if w1_mat.shape[0] == 0 or w2_mat.shape[0] == 0:
        return float("nan")

    if concept_centroid.size == 0:
        return float("nan")

    sampling = str(target_context_sampling or "observed").lower()
    if sampling not in {"observed", "balanced"}:
        raise ValueError("target_context_sampling must be 'observed' or 'balanced'")

    pooled = np.vstack([w1_mat, w2_mat]).astype(np.float32)
    n1 = int(w1_mat.shape[0])
    n2 = int(w2_mat.shape[0])
    n_total = int(pooled.shape[0])

    if n1 <= 0 or n2 <= 0 or n_total < 2:
        return float("nan")

    rng_local = np.random.default_rng(seed)
    ge = 0

    if sampling == "balanced":
        k = min(n1, n2)

        if k <= 0 or 2 * k > n_total:
            return float("nan")

        all_idx = np.arange(n_total)

        for _ in range(int(n_perm)):
            idx1 = rng_local.choice(n_total, size=k, replace=False)
            remaining = np.setdiff1d(all_idx, idx1, assume_unique=False)
            idx2 = rng_local.choice(remaining, size=k, replace=False)

            perm_w1 = pooled[idx1].mean(axis=0)
            perm_w2 = pooled[idx2].mean(axis=0)

            perm_diff = cosine(perm_w1, concept_centroid) - cosine(perm_w2, concept_centroid)

            if abs(perm_diff) >= abs(float(observed)):
                ge += 1

        return float((ge + 1) / (int(n_perm) + 1))

    if n1 >= n_total:
        return float("nan")

    for _ in range(int(n_perm)):
        idx1 = rng_local.choice(n_total, size=n1, replace=False)
        mask = np.zeros(n_total, dtype=bool)
        mask[idx1] = True

        perm_w1 = pooled[mask].mean(axis=0)
        perm_w2 = pooled[~mask].mean(axis=0)

        perm_diff = cosine(perm_w1, concept_centroid) - cosine(perm_w2, concept_centroid)

        if abs(perm_diff) >= abs(float(observed)):
            ge += 1

    return float((ge + 1) / (int(n_perm) + 1))

def collect_context_sentences(raw_sents: List[str], lemma_sents: List[List[str]], target_word: str, max_k: int) -> List[str]:
    """Pick up to max_k sentences that contain the target by lemma or tolerant surface match."""
    idxs = collect_context_indices(raw_sents, lemma_sents, target_word, max_k)
    return [raw_sents[j] for j in idxs]

def collect_context_indices_no_rng(
    raw_sents: List[str],
    lemma_sents: List[List[str]],
    target_word: str,
) -> List[int]:
    """
    Collect all sentence indices containing target_word without sampling.

    This is for context-audit exports only. It avoids the global rng so that
    adding audit output cannot change bootstrap results.
    """
    target = str(target_word).lower()

    idxs = [idx for idx, lemmas in enumerate(lemma_sents) if target in lemmas]

    if not idxs:
        idxs = [
            idx
            for idx, sentence in enumerate(raw_sents)
            if _find_term_spans(sentence, target)
        ]

    return sorted(set(int(i) for i in idxs))


def stable_sample_indices(
    indices: List[int],
    max_examples: int,
    salt: str,
) -> List[int]:
    """
    Deterministically sample context indices for audit display.

    Uses a local RNG seeded from a hash, not the pipeline rng.
    """
    clean = sorted(set(int(i) for i in indices))

    if len(clean) <= max_examples:
        return clean

    seed_int = int(hashlib.sha256(salt.encode("utf-8")).hexdigest(), 16) % (2 ** 32)
    local_rng = np.random.default_rng(seed_int)
    selected = local_rng.choice(clean, size=int(max_examples), replace=False)

    return sorted(int(i) for i in selected.tolist())


def append_context_audit_rows(
    audit_rows: List[Dict[str, Any]],
    raw_sents: List[str],
    lemma_sents: List[List[str]],
    mode_name: str,
    group_name: str,
    term1: str,
    term2: str,
    concept_name: str,
    concept_words: List[str],
    max_examples_per_item: int = 6,
) -> None:
    """
    Append readable context examples for one pair-concept result.

    Rows include target contexts for term1 and term2, plus contexts for each
    concept seed word. This is a qualitative audit supplement, not an input to
    the statistical test.
    """

    def _add_rows(context_role: str, context_term: str) -> None:
        all_idxs = collect_context_indices_no_rng(raw_sents, lemma_sents, context_term)

        salt = "|".join([
            str(SEED),
            str(mode_name),
            str(group_name),
            str(term1),
            str(term2),
            str(concept_name),
            str(context_role),
            str(context_term),
        ])

        sampled_idxs = stable_sample_indices(
            all_idxs,
            max_examples=max_examples_per_item,
            salt=salt,
        )

        for rank, sent_idx in enumerate(sampled_idxs, start=1):
            audit_rows.append({
                "mode": str(mode_name).upper(),
                "bootstrap_group": str(group_name),
                "term_1": str(term1),
                "term_2": str(term2),
                "concept": str(concept_name),
                "context_role": str(context_role),
                "context_term": str(context_term),
                "context_rank": int(rank),
                "sentence_index": int(sent_idx),
                "sentence": str(raw_sents[sent_idx]),
            })

    _add_rows("term_1", term1)
    _add_rows("term_2", term2)

    for seed_word in concept_words:
        _add_rows("concept_seed", str(seed_word))


def build_context_audit_master_csv() -> Optional[Path]:
    """
    Combine all per-mode/per-group context-audit CSVs into one master CSV.

    Keeps the individual files and additionally writes:
        results_paper3/<run_tag>/context_audit_master.csv
        results_paper3/<run_tag>/master/paper_context_audit_master.csv
    """
    audit_dir = RES / "context_audits"

    if not audit_dir.is_dir():
        LOGGER.warning("No context audit directory found: %s", audit_dir)
        return None

    audit_paths = sorted(
        p for p in audit_dir.glob("context_audit_*.csv")
        if not p.name.startswith("context_audit_master")
    )

    if not audit_paths:
        LOGGER.warning("No per-mode/per-group context audit CSVs found in %s", audit_dir)
        return None

    frames: List[pd.DataFrame] = []

    for audit_path in audit_paths:
        try:
            part = pd.read_csv(audit_path)
        except Exception as exc:
            LOGGER.warning("Could not read context audit CSV %s: %s", audit_path, exc)
            continue

        part.insert(0, "audit_source_file", audit_path.name)
        frames.append(part)

    if not frames:
        LOGGER.warning("No readable context audit CSVs found in %s", audit_dir)
        return None

    master_df = pd.concat(frames, ignore_index=True)

    out_run = RES / "context_audit_master.csv"
    master_df.to_csv(out_run, index=False, encoding="utf-8")

    MASTER.mkdir(parents=True, exist_ok=True)
    out_master = MASTER / "paper_context_audit_master.csv"
    master_df.to_csv(out_master, index=False, encoding="utf-8")

    LOGGER.info("Wrote context audit master CSV to %s", out_run)
    LOGGER.info("Wrote paper context audit master CSV to %s", out_master)

    return out_run

def bert_contextual_embedding(target_word: str, raw_sents: List[str], lemma_sents: List[List[str]],
                              model: BertForMaskedLM, tokenizer: BertTokenizerFast,
                              max_contexts: int, use_last: bool) -> Optional[np.ndarray]:
    """Average contextual embedding for a target across up to max_contexts sentences."""
    sents = collect_context_sentences(raw_sents, lemma_sents, target_word, max_contexts)
    if not sents:
        return None
    vecs: List[np.ndarray] = []
    t_low = target_word.lower()
    for s in sents:
        v = bert_sentence_token_embeddings(s, t_low, model, tokenizer, use_last=use_last)
        if v is not None:
            vecs.append(v)
    if not vecs:
        return None
    return np.mean(np.stack(vecs), axis=0)

def bert_alignment(target: str, concept_words: List[str],
                   raw_sents: List[str], lemma_sents: List[List[str]],
                   model: BertForMaskedLM, tokenizer: BertTokenizerFast,
                   max_contexts: int, use_last: bool) -> float:
    """Cosine(target, centroid(concepts)) using contextual BERT embeddings."""
    tv = bert_contextual_embedding(target, raw_sents, lemma_sents, model, tokenizer, max_contexts, use_last)
    if tv is None:
        return 0.0
    concept_vecs = []
    for w in concept_words:
        cv = bert_contextual_embedding(w, raw_sents, lemma_sents, model, tokenizer, max_contexts, use_last)
        if cv is not None:
            concept_vecs.append(cv)
    if not concept_vecs:
        return 0.0
    centroid = np.mean(np.stack(concept_vecs), axis=0)
    return cosine(tv, centroid)

def bert_bootstrap_diff(word1: str, word2: str,
                        raw_sents: List[str],
                        lemma_sents: List[List[str]],
                        concept_words: List[str], n_boot: int,
                        model: BertForMaskedLM, tokenizer: BertTokenizerFast,
                        max_contexts: int, use_last: bool) -> Tuple[float, float, float, List[float], Dict[str, int]]:
    """Bootstrap the difference in BERT alignments for word1 minus word2 (sentence bootstrap)."""
    a1 = bert_alignment(word1, concept_words, raw_sents, lemma_sents, model, tokenizer, max_contexts, use_last)
    a2 = bert_alignment(word2, concept_words, raw_sents, lemma_sents, model, tokenizer, max_contexts, use_last)
    observed = a1 - a2

    diffs: List[float] = []
    n = len(lemma_sents)
    used_counts = {"word1_contexts": 0, "word2_contexts": 0, "concept_contexts": 0}
    w1_ctx = collect_context_sentences(raw_sents, lemma_sents, word1, max_contexts); used_counts["word1_contexts"] = len(w1_ctx)
    w2_ctx = collect_context_sentences(raw_sents, lemma_sents, word2, max_contexts); used_counts["word2_contexts"] = len(w2_ctx)
    used_counts["concept_contexts"] = int(sum(len(collect_context_sentences(raw_sents, lemma_sents, w, max_contexts)) for w in concept_words))

    for _ in tqdm(range(n_boot), desc=f"BERT {word1} vs {word2}", leave=False):
        idxs = rng.integers(0, n, size=n)
        lemma_sample = [lemma_sents[i] for i in idxs]
        raw_sample = [raw_sents[i] for i in idxs]
        d = (bert_alignment(word1, concept_words, raw_sample, lemma_sample, model, tokenizer, max_contexts, use_last)
             - bert_alignment(word2, concept_words, raw_sample, lemma_sample, model, tokenizer, max_contexts, use_last))
        diffs.append(d)

    if not diffs:
        return observed, 0.0, 0.0, [], used_counts
    lo, hi = float(np.percentile(diffs, 2.5)), float(np.percentile(diffs, 97.5))
    return observed, lo, hi, diffs, used_counts

def fdr_bh(p_vals: Sequence[float]) -> List[float]:
    """Benjamini–Hochberg FDR correction with monotone adjustment."""
    p = np.asarray(p_vals, dtype=float)
    n = p.size
    if n == 0:
        return []
    order = np.argsort(p)
    p_sorted = p[order]
    ranks = np.arange(1, n + 1, dtype=float)
    q_sorted = (p_sorted * n) / ranks
    q_sorted = np.minimum.accumulate(q_sorted[::-1])[::-1]
    q = np.empty_like(q_sorted); q[order] = q_sorted
    return np.clip(q, 0.0, 1.0).tolist()

def holm_bonferroni(p_vals: Sequence[float]) -> List[float]:
    """Holm–Bonferroni step-down adjusted p-values (vectorised)."""
    p = np.asarray(p_vals, dtype=float)
    n = int(p.size)
    if n == 0:
        return []
    order = np.argsort(p)
    p_sorted = p[order]
    factors = (n - np.arange(n)).astype(float)
    adj_sorted = np.minimum(p_sorted * factors, 1.0)
    adj_sorted = np.maximum.accumulate(adj_sorted)
    adj = np.empty_like(adj_sorted); adj[order] = adj_sorted
    return adj.tolist()

def fdr_by(p_vals: Sequence[float]) -> List[float]:
    """Benjamini–Yekutieli FDR, robust to arbitrary dependence."""
    p = np.asarray(p_vals, dtype=float)
    n = p.size
    if n == 0:
        return []
    c_m = np.sum(1.0 / np.arange(1, n + 1))
    order = np.argsort(p)
    p_sorted = p[order]
    ranks = np.arange(1, n + 1, dtype=float)
    q_sorted = (p_sorted * n * c_m) / ranks
    q_sorted = np.minimum.accumulate(q_sorted[::-1])[::-1]
    q = np.empty_like(q_sorted); q[order] = q_sorted
    return np.clip(q, 0.0, 1.0).tolist()

def _summary_from_bootstrap(diffs: List[float], observed: float, perm_iters: int = 0) -> Dict[str, float]:
    """Compute rich summary stats and optional sign-flip permutation proxy p-value."""
    out: Dict[str, float] = {}
    if not diffs:
        keys = [
            "boot_mean","boot_median","boot_sd","boot_se","boot_min","boot_max",
            "q05","q16","q25","q50","q75","q84","q95",
            "skew","kurtosis_excess","p_boot_2sided","p_boot_greater","p_boot_less",
            "cliffs_delta","bias_boot","p_perm_signflip"
        ]
        for k in keys: out[k] = float("nan")
        return out

    x = np.asarray(diffs, dtype=float)
    n = x.size
    mean = float(np.mean(x))
    med  = float(np.median(x))
    sd   = float(np.std(x, ddof=1)) if n > 1 else 0.0
    se   = float(sd / np.sqrt(n)) if n > 1 else 0.0

    q05, q16, q25, q50, q75, q84, q95 = np.percentile(x, [5,16,25,50,75,84,95]).tolist()

    if sd > 0.0:
        xc = x - mean
        m3 = float(np.mean(xc**3))
        m4 = float(np.mean(xc**4))
        skew = float(m3 / (sd**3))
        kurt_ex = float(m4 / (sd**4) - 3.0)
    else:
        skew = 0.0
        kurt_ex = -3.0

    abs_obs = abs(observed)
    p2 = (int(np.sum(np.abs(x) >= abs_obs)) + 1) / (n + 1)
    pg = (int(np.sum(x >= observed)) + 1) / (n + 1)
    pl = (int(np.sum(x <= observed)) + 1) / (n + 1)

    pi_pos = float(np.mean(x > 0))
    pi_neg = float(np.mean(x < 0))
    cliffs = float(pi_pos - pi_neg)

    bias = mean - float(observed)

    p_perm = float("nan")
    if perm_iters and n > 0:
        centred = x - mean
        abs_obs_c = abs(observed - mean)
        rng_local = np.random.default_rng(
            int(hashlib.md5(f"{SEED}|{len(x)}|{observed}".encode()).hexdigest(), 16) % (2 ** 32))
        remain = int(perm_iters)
        ge = 0
        batch = min(4096, remain)
        while remain > 0:
            k = min(batch, remain)
            signs = rng_local.choice([-1.0, 1.0], size=(k, n))
            sims = np.mean(signs * centred, axis=1)
            ge += int(np.sum(np.abs(sims) >= abs_obs_c))
            remain -= k
        p_perm = (ge + 1) / (perm_iters + 1)

    out.update({
        "boot_mean": mean, "boot_median": med, "boot_sd": sd, "boot_se": se,
        "boot_min": float(np.min(x)), "boot_max": float(np.max(x)),
        "q05": q05, "q16": q16, "q25": q25, "q50": q50, "q75": q75, "q84": q84, "q95": q95,
        "skew": skew, "kurtosis_excess": kurt_ex,
        "p_boot_2sided": float(p2), "p_boot_greater": float(pg), "p_boot_less": float(pl),
        "cliffs_delta": cliffs, "bias_boot": bias, "p_perm_signflip": p_perm,
    })
    return out

def _term_key(value: Any) -> str:
    """Normalize a term for pair matching without changing display labels."""
    text = str(value).strip().lower()
    text = text.replace("ō", "o").replace("ū", "u")
    text = text.replace("shōji", "shoji").replace("shōji", "shoji")
    text = re.sub(r"[^a-z0-9]+", "", text)
    return text


def _same_pair(row: pd.Series, term1: str, term2: str) -> bool:
    """Return True when a row matches a pair in either spelling-normalized form."""
    return (
        _term_key(row.get("term_1", "")) == _term_key(term1)
        and _term_key(row.get("term_2", "")) == _term_key(term2)
    )


def _pair_display(term1: str, term2: str) -> str:
    """Readable pair label."""
    return f"{term1} / {term2}"


def _plain_label(value: Any) -> str:
    """Readable label for concepts, modes, and technical identifiers."""
    raw = str(value).strip()
    if raw in PAPER3_AXIS_LABELS:
        return PAPER3_AXIS_LABELS[raw]

    text = raw.replace("_", " ").replace("-", " ")
    text = re.sub(r"(?<=[a-z])(?=[A-Z])", " ", text)
    text = re.sub(r"\s+", " ", text).strip()
    return text[:1].upper() + text[1:]


def _concept_order(concepts: Sequence[str], family: str) -> List[str]:
    """Return concepts in a stable, paper-facing order."""
    base_order = (
        PAPER3_ARCHITECTURE_AXIS_ORDER
        if family == "architecture"
        else PAPER3_AESTHETIC_AXIS_ORDER
    )

    present = [c for c in base_order if c in set(concepts)]
    extras = sorted([c for c in concepts if c not in set(base_order)], key=_plain_label)
    return present + extras


def _concept_order_for_pair(term1: str, term2: str, concepts: Sequence[str]) -> List[str]:
    """Return the correct pair-specific concept order."""
    key = (_term_key(term1), _term_key(term2))
    available = set(str(c) for c in concepts)

    base_order = PAPER3_PAIR_CONCEPT_ORDER.get(key, [])
    present = [c for c in base_order if c in available]
    extras = sorted([c for c in available if c not in set(base_order)], key=_plain_label)

    return present + extras


def _num(value: Any) -> float:
    """Safe numeric conversion."""
    try:
        out = float(value)
    except (TypeError, ValueError):
        return float("nan")
    return out if np.isfinite(out) else float("nan")


def _ci_excludes_zero(low: Any, high: Any) -> bool:
    """Return True if the confidence interval is fully on one side of zero."""
    lo = _num(low)
    hi = _num(high)
    if not np.isfinite(lo) or not np.isfinite(hi):
        return False
    return bool(lo > 0.0 or hi < 0.0)

def _energy_distance(x: np.ndarray, y: np.ndarray) -> float:
    """Energy distance between two samples of shape (n,d) and (m,d)."""
    if x.size == 0 or y.size == 0:
        return float("nan")
    n = min(len(x), 512); m = min(len(y), 512)
    xi = x[:n]; yi = y[:m]
    d_xy = pairwise_distances(xi, yi).mean()
    d_xx = pairwise_distances(xi, xi).mean()
    d_yy = pairwise_distances(yi, yi).mean()
    return 2.0 * d_xy - d_xx - d_yy


def _mmd_rbf(x: np.ndarray, y: np.ndarray, gamma: float | None = None) -> float:
    """Unbiased MMD^2 with RBF kernel; gamma defaults to 1/feature_dim."""
    if x.size == 0 or y.size == 0:
        return float("nan")
    d = x.shape[1]
    g = (1.0 / d) if gamma is None else float(gamma)
    n = min(len(x), 512); m = min(len(y), 512)
    xi = x[:n]; yi = y[:m]
    def _k(a: np.ndarray, b: np.ndarray) -> np.ndarray:
        d2 = pairwise_distances(a, b, metric="sqeuclidean")
        return np.exp(-g * d2)
    k_xx = _k(xi, xi); k_yy = _k(yi, yi); k_xy = _k(xi, yi)
    # Unbiased estimates drop diagonal
    np.fill_diagonal(k_xx, 0.0); np.fill_diagonal(k_yy, 0.0)
    term_xx = k_xx.sum() / (n * (n - 1)) if n > 1 else 0.0
    term_yy = k_yy.sum() / (m * (m - 1)) if m > 1 else 0.0
    term_xy = k_xy.mean()
    return term_xx + term_yy - 2.0 * term_xy

def _silhouette_two_sample(x: np.ndarray, y: np.ndarray) -> float:
    """Silhouette on the union of two samples with labels {0,1} using euclidean distances."""
    if len(x) < 2 or len(y) < 2:
        return float("nan")
    z = np.vstack([x, y])
    labels = np.array([0] * len(x) + [1] * len(y))
    try:
        return float(silhouette_score(z, labels, metric="euclidean"))
    except (ValueError, RuntimeError):
        return float("nan")

def _row_normalise(mat: np.ndarray) -> np.ndarray:
    """L2-normalise rows; safe on empty input."""
    if mat.size == 0:
        return mat
    norms = np.linalg.norm(mat, axis=1, keepdims=True)
    norms[norms == 0.0] = 1.0
    return mat / norms

def _pairwise_cos(x: np.ndarray, y: np.ndarray) -> np.ndarray:
    """Pairwise cosine matrix between rows of x and y."""
    if x.size == 0 or y.size == 0:
        return np.empty((x.shape[0], y.shape[0]), dtype=np.float32)
    x_n = _row_normalise(x.astype(np.float32))
    y_n = _row_normalise(y.astype(np.float32))
    return x_n @ y_n.T

def _concept_centroid_from_pool(pre_pool: List[np.ndarray]) -> np.ndarray:
    """Equal-weight centroid across concept words, using each word's mean occurrence vector."""
    word_means = [p.mean(axis=0) for p in pre_pool if p.size]
    if not word_means:
        return np.zeros((768,), dtype=np.float32)
    return np.mean(np.stack(word_means, axis=0), axis=0).astype(np.float32)

def _bootstrap_concept_centroid_from_pool(pre_pool: List[np.ndarray]) -> np.ndarray:
    """Bootstrap an equal-weight concept centroid by resampling within each concept word."""
    word_means: List[np.ndarray] = []
    dim = 768
    for p in pre_pool:
        if p.size:
            dim = p.shape[1]
            word_means.append(_mean_sample(p, p.shape[0]))
    if not word_means:
        return np.zeros((dim,), dtype=np.float32)
    return np.mean(np.stack(word_means, axis=0), axis=0).astype(np.float32)

def _filter_by_concept_quantile(mat: np.ndarray, concept_centroid: np.ndarray, q: float) -> np.ndarray:
    """Keep rows whose cosine to the concept centroid is in the top quantile q."""
    if mat.size == 0:
        return mat
    if concept_centroid.size == 0:
        return mat
    cc = concept_centroid.reshape(1, -1).astype(np.float32)
    sims = _pairwise_cos(mat, cc).ravel()
    thresh = float(np.quantile(sims, q))
    keep = mat[sims >= thresh]
    return keep if keep.size else mat  # fall back to full set if all filtered out

def wpi_point_estimates(x: np.ndarray, y: np.ndarray) -> Tuple[float, float, float, int, int]:
    """
    Returns:
      centroid_cos: cosine(mean(x), mean(y))
      nn_overlap: 0.5*(mean max cos(x->y) + mean max cos(y->x))
      asym: mean max cos(x->y) - mean max cos(y->x)   (directionality)
      n_x, n_y: counts
    """
    n1, n2 = int(x.shape[0]), int(y.shape[0])
    if n1 == 0 or n2 == 0:
        return float("nan"), float("nan"), float("nan"), n1, n2
    c1 = x.mean(axis=0); c2 = y.mean(axis=0)
    centroid_cos = cosine(c1, c2)

    sims = _pairwise_cos(x, y)
    if sims.size == 0:
        return centroid_cos, float("nan"), float("nan"), n1, n2
    nn12 = float(np.mean(np.max(sims, axis=1)))
    nn21 = float(np.mean(np.max(sims, axis=0)))
    nn_overlap = 0.5 * (nn12 + nn21)
    asym = nn12 - nn21
    return centroid_cos, nn_overlap, asym, n1, n2

def wpi_bootstrap(
    x: np.ndarray, y: np.ndarray, b_iters: int, rng_local: np.random.Generator, obs_asym: float
) -> Dict[str, float]:
    """
    Bootstrap WPI metrics by resampling occurrences with replacement.
    Provides CIs and two-sided p for asym (null = 0) using the observed asymmetry.
    """
    n1, n2 = x.shape[0], y.shape[0]
    if n1 == 0 or n2 == 0 or b_iters <= 0:
        return {
            "wpi_nn_ci_low": float("nan"), "wpi_nn_ci_high": float("nan"),
            "wpi_asym_ci_low": float("nan"), "wpi_asym_ci_high": float("nan"),
            "wpi_asym_p_boot_2sided": float("nan")
        }
    nn_vals: List[float] = []
    asym_vals: List[float] = []
    for _ in range(b_iters):
        xi = x[rng_local.integers(0, n1, size=n1)]
        yi = y[rng_local.integers(0, n2, size=n2)]
        _, nn_i, asym_i, _, _ = wpi_point_estimates(xi, yi)
        if np.isfinite(nn_i): nn_vals.append(nn_i)
        if np.isfinite(asym_i): asym_vals.append(asym_i)
    if not nn_vals or not asym_vals:
        return {
            "wpi_nn_ci_low": float("nan"), "wpi_nn_ci_high": float("nan"),
            "wpi_asym_ci_low": float("nan"), "wpi_asym_ci_high": float("nan"),
            "wpi_asym_p_boot_2sided": float("nan")
        }
    nn_arr = np.asarray(nn_vals, dtype=float)
    asym_arr = np.asarray(asym_vals, dtype=float)
    nn_lo, nn_hi = np.percentile(nn_arr, [2.5, 97.5]).tolist()
    a_lo, a_hi   = np.percentile(asym_arr, [2.5, 97.5]).tolist()
    p2 = (int(np.sum(np.abs(asym_arr) >= abs(float(obs_asym)))) + 1) / (asym_arr.size + 1)
    return {
        "wpi_nn_ci_low": float(nn_lo), "wpi_nn_ci_high": float(nn_hi),
        "wpi_asym_ci_low": float(a_lo), "wpi_asym_ci_high": float(a_hi),
        "wpi_asym_p_boot_2sided": float(p2)
    }

def pairwise_similarity_point_estimates(x: np.ndarray, y: np.ndarray) -> Dict[str, float]:
    """
    Direct term-to-term similarity summaries.

    Returns:
      mean_cos: cosine(mean(x), mean(y))
      nn_mean_xy: mean max cosine from x to y
      nn_mean_yx: mean max cosine from y to x
      nn_mean_sym: symmetric mean of the two directions
    """
    n1, n2 = int(x.shape[0]), int(y.shape[0])
    if n1 == 0 or n2 == 0:
        return {
            "mean_cos": float("nan"),
            "nn_mean_xy": float("nan"),
            "nn_mean_yx": float("nan"),
            "nn_mean_sym": float("nan"),
        }

    mean_cos = cosine(x.mean(axis=0), y.mean(axis=0))
    sims = _pairwise_cos(x, y)
    if sims.size == 0:
        return {
            "mean_cos": mean_cos,
            "nn_mean_xy": float("nan"),
            "nn_mean_yx": float("nan"),
            "nn_mean_sym": float("nan"),
        }

    nn_xy = float(np.mean(np.max(sims, axis=1)))
    nn_yx = float(np.mean(np.max(sims, axis=0)))
    nn_sym = 0.5 * (nn_xy + nn_yx)
    return {
        "mean_cos": mean_cos,
        "nn_mean_xy": nn_xy,
        "nn_mean_yx": nn_yx,
        "nn_mean_sym": nn_sym,
    }

def pairwise_similarity_bootstrap(
    x: np.ndarray,
    y: np.ndarray,
    b_iters: int,
    rng_local: np.random.Generator,
) -> Dict[str, float]:
    """Bootstrap direct pairwise similarity summaries by resampling occurrences with replacement."""
    n1, n2 = int(x.shape[0]), int(y.shape[0])
    if n1 == 0 or n2 == 0 or b_iters <= 0:
        return {
            "pair_mean_cos_ci_low": float("nan"),
            "pair_mean_cos_ci_high": float("nan"),
            "pair_nn_xy_ci_low": float("nan"),
            "pair_nn_xy_ci_high": float("nan"),
            "pair_nn_yx_ci_low": float("nan"),
            "pair_nn_yx_ci_high": float("nan"),
            "pair_nn_sym_ci_low": float("nan"),
            "pair_nn_sym_ci_high": float("nan"),
        }

    cos_vals: List[float] = []
    nn_xy_vals: List[float] = []
    nn_yx_vals: List[float] = []
    nn_sym_vals: List[float] = []

    for _ in range(b_iters):
        xi = x[rng_local.integers(0, n1, size=n1)]
        yi = y[rng_local.integers(0, n2, size=n2)]
        est = pairwise_similarity_point_estimates(xi, yi)
        if np.isfinite(est["mean_cos"]):
            cos_vals.append(est["mean_cos"])
        if np.isfinite(est["nn_mean_xy"]):
            nn_xy_vals.append(est["nn_mean_xy"])
        if np.isfinite(est["nn_mean_yx"]):
            nn_yx_vals.append(est["nn_mean_yx"])
        if np.isfinite(est["nn_mean_sym"]):
            nn_sym_vals.append(est["nn_mean_sym"])

    if not cos_vals or not nn_xy_vals or not nn_yx_vals or not nn_sym_vals:
        return {
            "pair_mean_cos_ci_low": float("nan"),
            "pair_mean_cos_ci_high": float("nan"),
            "pair_nn_xy_ci_low": float("nan"),
            "pair_nn_xy_ci_high": float("nan"),
            "pair_nn_yx_ci_low": float("nan"),
            "pair_nn_yx_ci_high": float("nan"),
            "pair_nn_sym_ci_low": float("nan"),
            "pair_nn_sym_ci_high": float("nan"),
        }

    cos_lo, cos_hi = np.percentile(np.asarray(cos_vals, dtype=float), [2.5, 97.5]).tolist()
    xy_lo, xy_hi = np.percentile(np.asarray(nn_xy_vals, dtype=float), [2.5, 97.5]).tolist()
    yx_lo, yx_hi = np.percentile(np.asarray(nn_yx_vals, dtype=float), [2.5, 97.5]).tolist()
    sym_lo, sym_hi = np.percentile(np.asarray(nn_sym_vals, dtype=float), [2.5, 97.5]).tolist()

    return {
        "pair_mean_cos_ci_low": float(cos_lo),
        "pair_mean_cos_ci_high": float(cos_hi),
        "pair_nn_xy_ci_low": float(xy_lo),
        "pair_nn_xy_ci_high": float(xy_hi),
        "pair_nn_yx_ci_low": float(yx_lo),
        "pair_nn_yx_ci_high": float(yx_hi),
        "pair_nn_sym_ci_low": float(sym_lo),
        "pair_nn_sym_ci_high": float(sym_hi),
    }

def sanitise_metric_row(row: Dict[str, Any]) -> Dict[str, Any]:
    """Add validity flags and preserve unavailable metrics as NaN."""
    g_n1 = int(row.get("wpi_global_n_word1", 0) or 0)

    row["silhouette_concept_defined"] = bool(c_n1 >= min_sil and c_n2 >= min_sil)

        "pair_concept_nn_mean_sym_ci_low", "pair_concept_nn_mean_sym_ci_high",
        "energy_distance", "mmd_rbf", "silhouette_two_sample",
        "energy_distance_concept", "mmd_rbf_concept", "silhouette_two_sample_concept",
    ]

    for key in numeric_keys:
        val = row.get(key, np.nan)
        if val is None:
            row[key] = np.nan
            continue
        try:
            if not np.isfinite(float(val)):
                row[key] = np.nan
        except (TypeError, ValueError):
            row[key] = np.nan

    return row

def write_plot_availability_report(df: pd.DataFrame, out_dir: Path) -> None:
    """Write a CSV explaining which summary plots are plottable and why others are skipped."""
    rows: List[Dict[str, Any]] = []

    metric_specs = [
        ("heatmap_observed_difference.png", "observed_difference"),
        ("heatmap_pair_concept_mean_cos.png", "pair_concept_mean_cos"),
        ("heatmap_pair_concept_nn_mean_sym.png", "pair_concept_nn_mean_sym"),
    ]

    for figure_name, col in metric_specs:
        if col not in df.columns:
            rows.append({
                "figure": figure_name,
                "available": False,
                "reason": f"missing column: {col}",
                "finite_rows": 0,
            })
            continue

        finite_rows = int(df[col].notna().sum())
        rows.append({
            "figure": figure_name,
            "available": bool(finite_rows > 0),
            "reason": "ok" if finite_rows > 0 else f"no finite values in {col}",
            "finite_rows": finite_rows,
        })

    if "contexts_concept_total" in df.columns:
        finite_rows = int(df["contexts_concept_total"].notna().sum())
        rows.append({
            "figure": "bar_concept_anchor_coverage.png",
            "available": bool(finite_rows > 0),
            "reason": "ok" if finite_rows > 0 else "no concept coverage rows",
            "finite_rows": finite_rows,
        })
    else:
        rows.append({
            "figure": "bar_concept_anchor_coverage.png",
            "available": False,
            "reason": "missing column: contexts_concept_total",
            "finite_rows": 0,
        })

    if {"model", "term_1", "term_2"}.issubset(df.columns):
        for (model_name, t1, t2), pair_df in df.groupby(["model", "term_1", "term_2"], dropna=False):
            if {"observed_difference", "ci_95_low", "ci_95_high"}.issubset(pair_df.columns):
                valid_contrast_rows = int(
                    pair_df[["observed_difference", "ci_95_low", "ci_95_high"]]
                    .dropna()
                    .shape[0]
                )
            else:
                valid_contrast_rows = 0

            rows.append({
                "figure": f"contrast_forest::{model_name}::{t1}::{t2}",
                "available": bool(valid_contrast_rows > 0),
                "reason": "ok" if valid_contrast_rows > 0 else "no valid contrast rows",
                "finite_rows": valid_contrast_rows,
            })

            pair_ok = bool(pair_df["pair_global_defined"].iloc[0]) if "pair_global_defined" in pair_df.columns and not pair_df.empty else False

            if {"pair_concept_mean_cos", "pair_concept_nn_mean_sym"}.issubset(pair_df.columns):
                valid_profile_rows = pair_df.dropna(subset=["pair_concept_mean_cos", "pair_concept_nn_mean_sym"])
            else:
                valid_profile_rows = pd.DataFrame()

            concept_ok = len(valid_profile_rows) >= 2

            if pair_ok and concept_ok:
                reason = "ok"
            elif not pair_ok and not concept_ok:
                reason = "global pair undefined and fewer than two valid concept-conditioned rows"
            elif not pair_ok:
                reason = "global pair undefined"
            else:
                reason = "fewer than two valid concept-conditioned rows"

            rows.append({
                "figure": f"similarity_profile::{model_name}::{t1}::{t2}",
                "available": bool(pair_ok and concept_ok),
                "reason": reason,
                "finite_rows": int(len(valid_profile_rows)),
            })

    report = pd.DataFrame(rows)
    out_dir.mkdir(parents=True, exist_ok=True)
    report.to_csv(out_dir / "plot_availability_report.csv", index=False)

@torch.no_grad()
def collect_occurrence_embeddings_for_word(
    word: str,
    raw_sents: List[str],
    lemma_sents: List[List[str]],
    model: BertForMaskedLM,
    tokenizer: BertTokenizerFast,
    max_contexts: int,
    use_last: bool
) -> np.ndarray:
    """Return per-occurrence contextual embeddings for 'word' across up to max_contexts sentences."""
    sents = collect_context_sentences(raw_sents, lemma_sents, word, max_contexts)
    t = word.lower()
    vecs: List[np.ndarray] = []
    for s in sents:
        v = bert_sentence_token_embeddings(s, t, model, tokenizer, use_last=use_last)
        if v is not None:
            vecs.append(v)
    return np.stack(vecs, axis=0).astype(np.float32) if vecs else np.empty((0, 768), dtype=np.float32)

def plot_interchangeability(sims_xy: np.ndarray, sims_yx: np.ndarray, term1: str, term2: str,
                            concept: str, figures_dir: Path, mode: str, group_name: str) -> None:
    """Plot actual nearest-neighbour cosine values in both directions."""
    if sims_xy.size == 0 or sims_yx.size == 0:
        return

    if concept == "GLOBAL":
        context_label = "all retained occurrences"
    elif concept.endswith("_CONCEPT"):
        base_concept = concept.replace("_CONCEPT", "")
        context_label = f"concept-conditioned on {_plain_label(base_concept)}"
    else:
        context_label = _plain_label(concept)

    j1 = np.linspace(-0.08, 0.08, len(sims_xy)) if len(sims_xy) > 1 else np.array([0.0])
    j2 = np.linspace(-0.08, 0.08, len(sims_yx)) if len(sims_yx) > 1 else np.array([0.0])

    y1 = np.zeros(len(sims_xy), dtype=float) + j1
    y2 = np.ones(len(sims_yx), dtype=float) + j2

    fig = plt.figure(figsize=(10.0, 4.5))
    ax = fig.add_subplot(111)
    ax.scatter(sims_xy, y1, marker="o", label=f"{term1} → nearest {term2} (n={len(sims_xy)})")
    ax.scatter(sims_yx, y2, marker="s", label=f"{term2} → nearest {term1} (n={len(sims_yx)})")

    ax.set_yticks([0.0, 1.0])
    ax.set_yticklabels([f"{term1} → {term2}", f"{term2} → {term1}"])
    ax.set_xlabel("Nearest-neighbour cosine similarity")
    ax.set_title(f"Nearest-neighbour cosines for {term1} vs {term2}, {context_label} ({mode}, group={group_name})")
    ax.grid(True, linestyle="--", alpha=0.6)
    ax.legend()
    fig.tight_layout()

    out = figures_dir / f"wpi_nn_points_{_safe_tag(group_name)}_{_safe_tag(term1)}_vs_{_safe_tag(term2)}_{_safe_tag(concept)}_{mode.lower()}.png"
    _safe_savefig(fig, out)
    plt.close(fig)

def plot_metric_heatmap(df: pd.DataFrame, value_col: str, title: str, out_path: Path, centre_zero: bool = False) -> None:
    """Heatmap over pair × concept for a selected metric, keeping unavailable cells masked."""
    if value_col not in df.columns or df.empty:
        return

    plot_df = (
        df.assign(pair=df["term_1"].astype(str) + " vs " + df["term_2"].astype(str))
          .pivot_table(index="pair", columns="concept", values=value_col, aggfunc="first")
    )

    if plot_df.empty:
        return

    plot_df = plot_df.sort_index().sort_index(axis=1)
    arr = plot_df.astype(float).replace([np.inf, -np.inf], np.nan).to_numpy(dtype=float)
    masked = np.ma.masked_invalid(arr)

    fig = plt.figure(figsize=(max(7.0, 1.2 * arr.shape[1] + 2.0), max(4.0, 0.55 * arr.shape[0] + 2.0)))
    ax = fig.add_subplot(111)

    cmap = plt.get_cmap("coolwarm" if centre_zero else "viridis").copy()
    cmap.set_bad(color="#d9d9d9")

    finite = masked.compressed()
    if finite.size == 0:
        return

    if centre_zero:
        lim = float(np.max(np.abs(finite)))
        lim = lim if lim > 0.0 else 1.0
        im = ax.imshow(masked, aspect="auto", cmap=cmap, vmin=-lim, vmax=lim)
    else:
        vmin = float(np.min(finite))
        vmax = float(np.max(finite))
        if vmax <= vmin:
            vmax = vmin + 1e-6
        im = ax.imshow(masked, aspect="auto", cmap=cmap, vmin=vmin, vmax=vmax)

    ax.set_title(f"{title} (grey = unavailable)")
    ax.set_xticks(np.arange(arr.shape[1]))
    ax.set_xticklabels([_plain_label(c) for c in plot_df.columns.tolist()], rotation=45, ha="right")
    ax.set_yticks(np.arange(arr.shape[0]))
    ax.set_yticklabels(plot_df.index.tolist())

    for i in range(arr.shape[0]):
        for j in range(arr.shape[1]):
            if np.isfinite(arr[i, j]):
                txt = f"{arr[i, j]:.3f}" if abs(arr[i, j]) < 100 else f"{arr[i, j]:.0f}"
                ax.text(j, i, txt, ha="center", va="center", fontsize=8)

    fig.colorbar(im, ax=ax, fraction=0.046, pad=0.04)
    fig.tight_layout()

    _safe_savefig(fig, out_path)
    plt.close(fig)

def plot_contrast_forest(df_pair: pd.DataFrame, figures_dir: Path, mode: str, model_name: str,
                         term1: str, term2: str, group_name: str) -> None:
    """Horizontal CI plot for valid concept-alignment differences across concepts."""
    needed = {"concept", "observed_difference", "ci_95_low", "ci_95_high"}
    if not needed.issubset(df_pair.columns):
        return

    d = (
        df_pair[["concept", "observed_difference", "ci_95_low", "ci_95_high"]]
        .dropna(subset=["observed_difference", "ci_95_low", "ci_95_high"])
        .sort_values("observed_difference")
        .reset_index(drop=True)
    )

    if d.empty:
        return

    y = np.arange(len(d))
    vals = d["observed_difference"].astype(float).to_numpy()
    lo = d["ci_95_low"].astype(float).to_numpy()
    hi = d["ci_95_high"].astype(float).to_numpy()

    fig = plt.figure(figsize=(10.0, max(4.0, 0.7 * len(d) + 1.0)))
    ax = fig.add_subplot(111)
    ax.axvline(0.0, linestyle="--", linewidth=1.5)
    ax.errorbar(
        vals, y,
        xerr=np.vstack([vals - lo, hi - vals]),
        fmt="o",
        capsize=4,
    )
    ax.set_yticks(y)
    ax.set_yticklabels([_plain_label(c) for c in d["concept"].astype(str).tolist()])
    ax.set_xlabel(f"Observed difference in concept alignment ({term1} minus {term2})")
    ax.set_title(f"{model_name}: concept-alignment contrast for {term1} vs {term2} ({mode}, group={group_name})")
    ax.grid(True, linestyle="--", alpha=0.6)
    fig.tight_layout()

    out = figures_dir / f"contrast_forest_{_safe_tag(group_name)}_{_safe_tag(term1)}_vs_{_safe_tag(term2)}_{mode.lower()}.png"
    _safe_savefig(fig, out)
    plt.close(fig)

def plot_similarity_profile(df_pair: pd.DataFrame, figures_dir: Path, mode: str, model_name: str,
                            term1: str, term2: str, group_name: str) -> None:
    """Concept-wise profile for valid pairwise similarity metrics."""
    needed = {
        "concept",
        "pair_concept_mean_cos", "pair_concept_nn_mean_sym",
        "pair_global_mean_cos", "pair_global_nn_mean_sym"
    }
    if not needed.issubset(df_pair.columns):
        return

    d = (
        df_pair[["concept", "pair_concept_mean_cos", "pair_concept_nn_mean_sym", "pair_global_mean_cos", "pair_global_nn_mean_sym"]]
        .dropna(subset=["pair_concept_mean_cos", "pair_concept_nn_mean_sym"])
        .sort_values("concept")
        .reset_index(drop=True)
    )

    if d.empty:
        return

    y = np.arange(len(d), dtype=float)
    y_mean = y - 0.12
    y_nn = y + 0.12

    c_mean = d["pair_concept_mean_cos"].astype(float).to_numpy()
    c_nn = d["pair_concept_nn_mean_sym"].astype(float).to_numpy()

    g_mean = float(d["pair_global_mean_cos"].iloc[0])
    g_nn = float(d["pair_global_nn_mean_sym"].iloc[0])

    fig = plt.figure(figsize=(10.0, max(4.0, 0.7 * len(d) + 1.0)))
    ax = fig.add_subplot(111)

    ax.scatter(c_mean, y_mean, marker="o", label=f"Pair mean cosine: {term1} vs {term2}")
    ax.scatter(c_nn, y_nn, marker="s", label=f"Pair NN symmetry: {term1} vs {term2}")

    for xv, yv in zip(c_mean, y_mean):
        ax.text(xv + 0.002, yv, f"{xv:.2f}", va="center", fontsize=8)
    for xv, yv in zip(c_nn, y_nn):
        ax.text(xv + 0.002, yv, f"{xv:.2f}", va="center", fontsize=8)

    ax.axvline(g_mean, linestyle="--", linewidth=1.5, label=f"Global pair mean cosine = {g_mean:.3f}")
    ax.axvline(g_nn, linestyle=":", linewidth=1.5, label=f"Global pair NN symmetry = {g_nn:.3f}")

    ax.set_yticks(y)
    ax.set_yticklabels([_plain_label(c) for c in d["concept"].astype(str).tolist()])
    ax.set_xlabel("Pairwise similarity between the two target terms")
    ax.set_title(f"{model_name}: pairwise similarity profile for {term1} vs {term2} ({mode}, group={group_name})")
    ax.grid(True, linestyle="--", alpha=0.6)
    ax.legend()
    fig.tight_layout()

    out = figures_dir / f"similarity_profile_{_safe_tag(group_name)}_{_safe_tag(term1)}_vs_{_safe_tag(term2)}_{mode.lower()}.png"
    _safe_savefig(fig, out)
    plt.close(fig)


def _publication_pair_subset(df: pd.DataFrame, term1: str, term2: str) -> pd.DataFrame:
    """Return rows for one declared pair."""
    if df.empty or not {"term_1", "term_2"}.issubset(df.columns):
        return pd.DataFrame()
    mask = df.apply(lambda row: _same_pair(row, term1, term2), axis=1)
    return df.loc[mask].copy()


def _mode_pair_summary(df_pair: pd.DataFrame, mode_name: str) -> str:
    """Compact FULL/CLEAN summary for the outcome matrix."""
    if df_pair.empty or "mode" not in df_pair.columns:
        return "unavailable"

    d = df_pair[df_pair["mode"].astype(str).str.upper().eq(mode_name.upper())].copy()
    if d.empty:
        return "unavailable"

    needed = ["observed_difference", "ci_95_low", "ci_95_high"]
    if not set(needed).issubset(d.columns):
        return "unavailable"

    d = d.dropna(subset=needed)
    if d.empty:
        return "unavailable"

    stable = int(
        d.apply(lambda row: _row_stable(row), axis=1).sum()
    )
    total = int(len(d))

    if stable == 0:
        return f"weak / no stable axes ({total} tested)"
    if stable == total:
        return f"stable on all axes ({stable}/{total})"
    return f"axis-specific ({stable}/{total} stable)"


def plot_publication_outcome_matrix(df: pd.DataFrame, out_path: Path) -> None:
    """
    Figure 1. Plain-language outcome matrix for architectural readers.

    This is a reading map, not the statistical proof.
    """
    if df.empty:
        return

    def _wrap_cell(text: Any, width: int) -> str:
        clean = re.sub(r"\s+", " ", str(text)).strip()
        return "\n".join(
            textwrap.wrap(
                clean,
                width=width,

def plot_publication_effects(
    df: pd.DataFrame,
    pair_specs: List[Tuple[str, str, str]],
    out_path: Path,
    title: str,
) -> None:
    """
    Integrated FULL/CLEAN dot-and-interval plot.

    Each pair receives only its own configured concept axes. The displayed x-axis
    is flipped relative to the stored statistic so that the left side of the plot
    corresponds to the first term in the panel title, and the right side corresponds
    to the second term.

    Numeric labels report the plotted directional effect, Δplot = term2 - term1.
    The CSV output remains unchanged and stores observed_difference as term1 - term2.
    """
    needed = {"mode", "term_1", "term_2", "concept", "observed_difference", "ci_95_low", "ci_95_high"}
    if df.empty or not needed.issubset(df.columns):
        return

    d_all: List[pd.DataFrame] = []

    for term1, term2, role in pair_specs:
        d_pair = _publication_pair_subset(df, term1, term2)
        if not d_pair.empty:
            d_pair = d_pair.copy()
            d_pair["_pub_term1"] = term1
            d_pair["_pub_term2"] = term2
            d_pair["_pub_role"] = role
            d_all.append(d_pair)

    if not d_all:
        return

    plot_df = pd.concat(d_all, ignore_index=True)
    plot_df = plot_df.dropna(subset=["observed_difference", "ci_95_low", "ci_95_high"])
    if plot_df.empty:
        return

    plot_df["_mode_norm"] = plot_df["mode"].astype(str).str.upper()
    plot_df = plot_df[plot_df["_mode_norm"].isin(["FULL", "CLEAN"])].copy()
    if plot_df.empty:
        return

    display_bounds: List[float] = []
    for _, row in plot_df.iterrows():
        val = _num(row.get("observed_difference"))
        lo = _num(row.get("ci_95_low"))
        hi = _num(row.get("ci_95_high"))
        if all(np.isfinite(x) for x in [val, lo, hi]):
            display_bounds.extend([-val, -hi, -lo])

    if not display_bounds:
        return

    max_abs = float(np.max(np.abs(np.asarray(display_bounds, dtype=float))))
    max_abs = max(max_abs, 0.01)

    xlim = (-max_abs * 1.34, max_abs * 1.34)
    label_pad = max_abs * 0.035

    pair_concept_lookup: Dict[Tuple[str, str], List[str]] = {}
    for term1, term2, _role in pair_specs:
        d_pair = plot_df[
            plot_df["_pub_term1"].eq(term1)
            & plot_df["_pub_term2"].eq(term2)
        ].copy()

        pair_concept_lookup[(term1, term2)] = _concept_order_for_pair(
            term1,
            term2,
            d_pair["concept"].astype(str).dropna().unique().tolist(),
        )

    max_concepts = max([len(v) for v in pair_concept_lookup.values()] + [1])
    n_pairs = len(pair_specs)

    fig, axes_raw = plt.subplots(
        1,
        n_pairs,
        figsize=(max(11.6, 5.7 * n_pairs), max(5.8, 0.66 * max_concepts + 2.45)),
        sharex=True,
        sharey=False,
    )
    fig.patch.set_facecolor("white")

    axes: List[Any] = list(np.asarray(axes_raw, dtype=object).reshape(-1))

    mode_styles = {
        "FULL": {
            "offset": -0.12,
            "marker": "o",
            "color": "#1f77b4",
        },
        "CLEAN": {
            "offset": 0.12,
            "marker": "s",
            "color": "#ff7f0e",
        },
    }

    for ax, (term1, term2, role) in zip(axes, pair_specs):
        d_pair = plot_df[
            plot_df["_pub_term1"].eq(term1)
            & plot_df["_pub_term2"].eq(term2)
        ].copy()

        concepts = pair_concept_lookup.get((term1, term2), [])
        y_base = np.arange(len(concepts), dtype=float)

        ax.set_facecolor("white")
        ax.axvline(0.0, color="#b22222", linestyle="--", linewidth=1.15, alpha=0.95)

        for mode_name in ["FULL", "CLEAN"]:
            style = mode_styles[mode_name]
            offset = float(style["offset"])
            marker = str(style["marker"])
            color = str(style["color"])

            for y_i, concept in zip(y_base, concepts):
                rows = d_pair[
                    d_pair["_mode_norm"].eq(mode_name)
                    & d_pair["concept"].astype(str).eq(concept)
                ]

                if rows.empty:
                    continue

                row = rows.iloc[0]
                val = _num(row["observed_difference"])
                lo = _num(row["ci_95_low"])
                hi = _num(row["ci_95_high"])

                if not all(np.isfinite(x) for x in [val, lo, hi]):
                    continue

                display_val = -val
                display_lo = -hi
                display_hi = -lo

                stable = _ci_excludes_zero(lo, hi)
                markerface = color if stable else "white"

                ax.errorbar(
                    display_val,
                    y_i + offset,
                    xerr=np.array([[max(0.0, display_val - display_lo)], [max(0.0, display_hi - display_val)]]),
                    fmt=marker,
                    capsize=3,
                    markersize=6.0,
                    markerfacecolor=markerface,
                    markeredgecolor=color,
                    ecolor=color,
                    color=color,
                    linewidth=1.25,
                )

                if display_val < 0:
                    text_x = display_lo - label_pad
                    text_ha = "right"
                else:
                    text_x = display_hi + label_pad
                    text_ha = "left"

                ax.text(
                    text_x,
                    y_i + offset,
                    f"{display_val:+.3f}",
                    ha=text_ha,
                    va="center",
                    fontsize=7.2,
                    color=color,
                    clip_on=True,
                )

        ax.set_title(f"{term1} / {term2}\n{role}", fontsize=10)
        ax.set_xlim(*xlim)
        ax.set_xlabel(f"← {term1} stronger     Δplot     {term2} stronger →", fontsize=8.6)
        ax.set_yticks(y_base)
        ax.set_yticklabels([_plain_label(c) for c in concepts])
        ax.invert_yaxis()
        ax.grid(True, axis="x", linestyle="--", alpha=0.35)
        ax.grid(True, axis="y", linestyle=":", alpha=0.20)

    fig.suptitle(title, fontsize=14, y=0.985)

    legend_handles = [
        Line2D([0], [0], marker="o", color="none", markerfacecolor="#1f77b4",
               markeredgecolor="#1f77b4", markersize=6, label="FULL"),
        Line2D([0], [0], marker="s", color="none", markerfacecolor="#ff7f0e",
               markeredgecolor="#ff7f0e", markersize=6, label="CLEAN"),
        Line2D([0], [0], marker="o", color="none", markerfacecolor="0.25",
               markeredgecolor="0.25", markersize=6, label="CI excludes zero"),
        Line2D([0], [0], marker="o", color="none", markerfacecolor="white",
               markeredgecolor="0.25", markersize=6, label="CI crosses zero"),
        Line2D([0], [0], color="#b22222", linestyle="--", linewidth=1.15,
               label="Zero-reference line"),
    ]

    fig.legend(
        handles=legend_handles,
        loc="lower center",
        ncol=5,
        frameon=False,
        bbox_to_anchor=(0.5, 0.018),
        fontsize=8.6,
    )

    fig.tight_layout(rect=(0.0, 0.10, 1.0, 0.94))
    _safe_savefig(fig, out_path)
    plt.close(fig)


def plot_publication_coverage_audit(df: pd.DataFrame, out_path: Path) -> None:
    """
    Pair-level coverage audit.

    The paper-facing question is whether each pair had enough target-term support,
    and how much CLEAN reduced it.
    """
    if df.empty:
        return

    required_any = {"mode", "term_1", "term_2"}
    if not required_any.issubset(df.columns):
        return

    term_rows: List[Dict[str, Any]] = []

    for _, row in df.iterrows():
        mode_name = str(row.get("mode", "")).upper()

        if "word1_contexts" in df.columns:
            term_rows.append({
                "mode": mode_name,
                "term": str(row.get("term_1", "")),
                "term_key": _term_key(row.get("term_1", "")),
                "contexts": _num(row.get("word1_contexts")),
            })

        if "word2_contexts" in df.columns:
            term_rows.append({
                "mode": mode_name,
                "term": str(row.get("term_2", "")),
                "term_key": _term_key(row.get("term_2", "")),
                "contexts": _num(row.get("word2_contexts")),
            })

    term_df = pd.DataFrame(term_rows)
    term_df = term_df.replace([np.inf, -np.inf], np.nan).dropna(subset=["term_key", "contexts"])

    if term_df.empty:
        return

    term_df = (
        term_df.groupby(["mode", "term_key"], as_index=False)
        .agg({"contexts": "max"})
        .reset_index(drop=True)
    )

    count_lookup: Dict[Tuple[str, str], float] = {}
    for rec in term_df[["mode", "term_key", "contexts"]].itertuples(index=False):
        count_lookup[(str(rec.mode).upper(), str(rec.term_key))] = float(rec.contexts)

    panel_specs: List[Tuple[str, List[Tuple[str, str, str]]]] = [
        ("Cross-cultural architectural pairs", PAPER3_CROSS_CULTURAL_PAIR_SPECS),
        ("Within-culture architectural pairs", PAPER3_WITHIN_CULTURE_PAIR_SPECS),
        ("Conceptual pairs", PAPER3_CONCEPTUAL_PAIR_SPECS),
    ]

    fig, axes_raw = plt.subplots(
        len(panel_specs),
        1,
        figsize=(12.5, 9.2),
        sharex=True,
        gridspec_kw={"height_ratios": [1.0, 1.0, 1.0]},
    )
    fig.patch.set_facecolor("white")

    axes: List[Any] = list(np.asarray(axes_raw, dtype=object).reshape(-1))

    mode_styles = {
        "FULL": {
            "offset": -0.17,
            "color": "#1f77b4",
        },
        "CLEAN": {
            "offset": 0.17,
            "color": "#ff7f0e",
        },
    }

    for ax, (panel_title, specs) in zip(axes, panel_specs):
        ax.set_facecolor("white")

        labels: List[str] = []
        y_positions: List[float] = []
        full_values: List[float] = []
        clean_values: List[float] = []

        y = 0.0
        separator_positions: List[float] = []

        for term1, term2, _role in specs:
            for term in [term1, term2]:
                term_key = _term_key(term)
                labels.append(term)
                y_positions.append(y)
                full_values.append(count_lookup.get(("FULL", term_key), 0.0))
                clean_values.append(count_lookup.get(("CLEAN", term_key), 0.0))
                y += 1.0

            separator_positions.append(y - 0.5)
            y += 0.45

        y_arr = np.asarray(y_positions, dtype=float)

        ax.barh(
            y_arr + float(mode_styles["FULL"]["offset"]),
            full_values,
            height=0.28,
            color=str(mode_styles["FULL"]["color"]),
            label="FULL",
        )
        ax.barh(
            y_arr + float(mode_styles["CLEAN"]["offset"]),
            clean_values,
            height=0.28,
            color=str(mode_styles["CLEAN"]["color"]),
            label="CLEAN",
        )

        for sep in separator_positions[:-1]:
            ax.axhline(sep, color="0.82", linewidth=0.8)

        ax.set_yticks(y_arr)
        ax.set_yticklabels(labels)
        ax.invert_yaxis()
        ax.set_title(panel_title, fontsize=11)
        ax.grid(True, axis="x", linestyle="--", alpha=0.42)

    axes[-1].set_xlabel("Retained target-term contexts")
    axes[0].legend(frameon=False, loc="lower right")

    fig.suptitle(
        "Pair-level corpus support in broad online encyclopedic discourse",
        fontsize=14,
        y=0.985,
    )

    fig.tight_layout(rect=(0.0, 0.035, 1.0, 0.95))
    _safe_savefig(fig, out_path)
    plt.close(fig)

def _fixed_paper_pair_specs() -> List[Tuple[str, str, str, str]]:
    """Return the fixed Paper 3 pair battery as (family, term1, term2, role)."""
    return (
        [("architecture", t1, t2, role) for t1, t2, role in PAPER3_ARCHITECTURE_PAIR_SPECS]
        + [("aesthetics", t1, t2, role) for t1, t2, role in PAPER3_AESTHETIC_PAIR_SPECS]
    )


def _paper_pair_key(term1: Any, term2: Any) -> Tuple[str, str]:
    """Canonical pair key."""
    return _term_key(term1), _term_key(term2)


def _paper_pair_metadata(term1: Any, term2: Any) -> Tuple[str, str]:
    """Return comparison family and pair type for the fixed Paper 3 battery."""
    key = _paper_pair_key(term1, term2)

    for family, spec_term1, spec_term2, role in _fixed_paper_pair_specs():
        if key == _paper_pair_key(spec_term1, spec_term2):
            return family, role

    return "other", "not in fixed paper battery"


def _paper_directional_reading(row: pd.Series) -> str:
    """Plain-language directional reading for the paper CSV."""
    term1 = str(row.get("term_1", "")).strip()
    term2 = str(row.get("term_2", "")).strip()
    concept = _plain_label(row.get("concept", ""))

    val = _num(row.get("observed_difference"))
    lo = _num(row.get("ci_95_low"))
    hi = _num(row.get("ci_95_high"))

    if not all(np.isfinite(x) for x in [val, lo, hi]):
        return "unavailable"

    if not _ci_excludes_zero(lo, hi):
        return "CI crosses zero; weak or unstable on this axis"

    if val > 0:
        return f"{term1} stronger on {concept}"

    if val < 0:
        return f"{term2} stronger on {concept}"

    return "no directional difference"


def _mode_row_for_pair_concept(
    df: pd.DataFrame,
    term1: str,
    term2: str,
    concept: str,
    mode_name: str,
) -> Optional[pd.Series]:
    """Return the row for a pair, concept, and mode, if present."""
    if df.empty:
        return None

    d_pair = _publication_pair_subset(df, term1, term2)
    if d_pair.empty:
        return None

    d_pair = d_pair.copy()
    d_pair["_mode_norm"] = d_pair["mode"].astype(str).str.upper()

    rows = d_pair[
        d_pair["_mode_norm"].eq(mode_name.upper())
        & d_pair["concept"].astype(str).eq(concept)
    ]

    if rows.empty:
        return None

    return rows.iloc[0]


def _row_direction(row: Optional[pd.Series]) -> str:
    """Return first_term, second_term, or empty based on observed difference."""
    if row is None:
        return ""

    val = _num(row.get("observed_difference"))
    if not np.isfinite(val):
        return ""

    if val > 0:
        return "first_term"
    if val < 0:
        return "second_term"
    return ""


def _row_stable(row: Optional[pd.Series]) -> bool:
    """
    Return True when a row meets the paper-facing strong-support rule.

    Strong support requires:
      1) the 95% bootstrap confidence interval excludes zero; and
      2) when BH-FDR correction is active, p_value_fdr_bh < 0.05.

    If BH-FDR is active but qBH is missing, the row is not labelled strong.
    """
    if row is None:
        return False

    ci_ok = _ci_excludes_zero(row.get("ci_95_low"), row.get("ci_95_high"))
    if not ci_ok:
        return False

    if bool(getattr(P3, "use_fdr_for_publication_status", True)):
        if "p_value_fdr_bh" not in row.index:
            return False

        q = _num(row.get("p_value_fdr_bh"))
        if not np.isfinite(q):
            return False

        return bool(q < 0.05)

    return True


def _row_ci_only(row: Optional[pd.Series]) -> bool:
    """
    Return True when the 95% CI excludes zero but the stricter corrected-audit
    rule does not mark the row as stable.

    This separates visible CI-level direction from publication-stable corrected
    audit status.
    """
    if row is None:
        return False

    ci_ok = _ci_excludes_zero(row.get("ci_95_low"), row.get("ci_95_high"))
    if not ci_ok:
        return False

    return not _row_stable(row)


def _term_for_direction(direction: str, term1: str, term2: str) -> str:
    """Convert first_term / second_term direction into the displayed term label."""
    if direction == "first_term":
        return term1

    if direction == "second_term":
        return term2

    return "directional"


def _stability_class_for_pair_concept(
    df: pd.DataFrame,
    term1: str,
    term2: str,
    concept: str,
) -> Tuple[str, str]:
    """
    Return (stability_class, stable_direction).

    stable_direction is first_term, second_term, or empty.
    """
    full_row = _mode_row_for_pair_concept(df, term1, term2, concept, "FULL")
    clean_row = _mode_row_for_pair_concept(df, term1, term2, concept, "CLEAN")

    full_stable = _row_stable(full_row)
    clean_stable = _row_stable(clean_row)

    full_direction = _row_direction(full_row)
    clean_direction = _row_direction(clean_row)

    pair_key = _paper_pair_key(term1, term2)

    if full_stable and clean_stable:
        if full_direction and clean_direction and full_direction != clean_direction:
            return "conflicting_direction", ""
        return "stable_both", full_direction or clean_direction

    if full_stable and not clean_stable:
        return "full_only", full_direction

    if clean_stable and not full_stable:
        return "clean_only", clean_direction

    if pair_key == _paper_pair_key("wabi", "sabi"):
        return "near_fusion_or_tight_coupling", ""

    if pair_key == _paper_pair_key("tokonoma", "chigaidana"):
        return "low_support_or_restraint", ""

    return "unstable_both", ""


def _stability_symbol_and_color(stability_class: str, stable_direction: str) -> Tuple[str, str]:
    """Return display symbol and background color for the stability map."""
    suffix = ""
    if stable_direction == "first_term":
        suffix = "1"
    elif stable_direction == "second_term":
        suffix = "2"

    if stability_class == "stable_both":
        return f"●{suffix}", "#d9ead3"
    if stability_class == "full_only":
        return f"F{suffix}", "#fff2cc"
    if stability_class == "clean_only":
        return f"C{suffix}", "#fff2cc"
    if stability_class == "conflicting_direction":
        return "×", "#f4cccc"
    if stability_class == "near_fusion_or_tight_coupling":
        return "≈", "#d9eaf7"
    if stability_class == "low_support_or_restraint":
        return "!", "#eadcf8"
    if stability_class == "unstable_both":
        return "○", "#f2f2f2"
    return "", "#ffffff"


def _draw_stability_panel(
    ax: Any,
    df: pd.DataFrame,
    pair_specs: List[Tuple[str, str, str]],
    concepts: List[str],
    title: str,
) -> None:
    """Draw one panel of the FULL/CLEAN stability map."""
    n_rows = len(pair_specs)
    n_cols = len(concepts)

    ax.set_xlim(0, n_cols)
    ax.set_ylim(0, n_rows)
    ax.set_aspect("equal")
    ax.invert_yaxis()

    for row_i, (term1, term2, _role) in enumerate(pair_specs):
        for col_i, concept in enumerate(concepts):
            stability_class, stable_direction = _stability_class_for_pair_concept(df, term1, term2, concept)
            symbol, color = _stability_symbol_and_color(stability_class, stable_direction)

            rect = Rectangle(
                (col_i, row_i),
                1,
                1,
                facecolor=color,
                edgecolor="0.65",
                linewidth=0.8,
            )
            ax.add_patch(rect)

            ax.text(
                col_i + 0.5,
                row_i + 0.5,
                symbol,
                ha="center",
                va="center",
                fontsize=12,
                fontweight="bold",
            )

    ax.set_xticks(np.arange(n_cols) + 0.5)
    ax.set_xticklabels([_plain_label(c) for c in concepts], rotation=45, ha="right")
    ax.set_yticks(np.arange(n_rows) + 0.5)
    ax.set_yticklabels([_pair_display(t1, t2) for t1, t2, _role in pair_specs])
    ax.tick_params(length=0)
    ax.set_title(title, fontsize=12, pad=12)

    for spine in ax.spines.values():
        spine.set_visible(False)


def plot_publication_stability_map(df: pd.DataFrame, out_path: Path) -> None:
    """
    Figure 3. Symbolic FULL/CLEAN stability map.

    This replaces numeric heatmaps as the paper-facing overview.
    """
    if df.empty:
        return

    fig, axes = plt.subplots(
        2,
        1,
        figsize=(12.0, 8.6),

    _draw_stability_panel(
        axes[1],
        df,

def _direction_text_for_row(row: Optional[pd.Series], term1: str, term2: str) -> str:
    """
    Return only the observed semantic leaning.

    This intentionally does not encode statistical support.
    """
    if row is None:
        return "not tested"

    direction = _row_direction(row)

    if not direction:
        return "none"

    return _term_for_direction(direction, term1, term2)


def _evidence_tier_text(row: Optional[pd.Series]) -> str:
    """
    Return the short paper-facing support label.

    strong:
        95% CI excludes zero and, when BH-FDR is available, qBH < .05.

    suggestive:
        95% CI excludes zero, but the BH-FDR threshold is not met.

    no clear split:
        95% CI crosses zero.
    """
    if row is None:
        return "not tested"

    if _row_stable(row):
        return "strong"

    if _row_ci_only(row):
        return "suggestive"

    return "no clear split"


def _support_cell_for_row(row: Optional[pd.Series], term1: str, term2: str) -> str:
    """
    Return the compact support cell for the main paper table.

    Examples:
        mu, strong
        mu, suggestive
        no clear split
    """
    support = _evidence_tier_text(row)

    if support in {"not tested", "no clear split"}:
        return support

    leaning = _direction_text_for_row(row, term1, term2)

    if leaning in {"not tested", "none"}:
        return support

    return f"{leaning}, {support}"


def _format_p_for_table(value: Any) -> str:
    """Format p/q values compactly for appendix tables."""
    val = _num(value)

    if not np.isfinite(val):
        return ""

    if val < 0.001:
        return "<.001"

    return f"{val:.3f}".lstrip("0")


def _ci_status_text(row: Optional[pd.Series]) -> str:
    """
    Return the CI status used to justify the support label.

    This uses ci_95_low and ci_95_high from paper_master_results.csv.
    """
    if row is None:
        return "not tested"

    lo = _num(row.get("ci_95_low"))
    hi = _num(row.get("ci_95_high"))

    if not np.isfinite(lo) or not np.isfinite(hi):
        return "not available"

    if _ci_excludes_zero(lo, hi):
        return "excludes 0"

    return "crosses 0"


def _p_perm_text(row: Optional[pd.Series]) -> str:
    """
    Return p_perm from p_perm_label in paper_master_results.csv.
    """
    if row is None:
        return ""

    return _format_p_for_table(row.get("p_perm_label"))


def _q_bh_text(row: Optional[pd.Series]) -> str:
    """
    Return qBH from p_value_fdr_bh in paper_master_results.csv.
    """
    if row is None:
        return ""

    return _format_p_for_table(row.get("p_value_fdr_bh"))


def _mode_status_text(row: Optional[pd.Series], term1: str, term2: str) -> str:
    """
    Backward-compatible compact mode label.

    The main table now uses _support_cell_for_row().
    """
    return _support_cell_for_row(row, term1, term2)


def _combined_status_text(
    full_row: Optional[pd.Series],
    clean_row: Optional[pd.Series],
    term1: str,
    term2: str,
) -> str:
    """
    Return a concise architectural reading for the main paper table.

    This avoids p-values and keeps the statistical detail in the appendix table.
    """
    full_support = _evidence_tier_text(full_row)
    clean_support = _evidence_tier_text(clean_row)

    full_supported = full_support in {"strong", "suggestive"}
    clean_supported = clean_support in {"strong", "suggestive"}

    full_leaning = _direction_text_for_row(full_row, term1, term2)
    clean_leaning = _direction_text_for_row(clean_row, term1, term2)

    if full_supported and clean_supported:
        if full_leaning != clean_leaning:
            return f"Direction changes: FULL leans {full_leaning}; CLEAN leans {clean_leaning}."

        if full_support == clean_support:
            return f"Consistent {full_leaning}-leaning; {full_support} support in both."

        if full_support == "strong" and clean_support == "suggestive":
            return f"Consistent {full_leaning}-leaning; stronger support in FULL."

        if clean_support == "strong" and full_support == "suggestive":
            return f"Consistent {full_leaning}-leaning; stronger support in CLEAN."

        return f"Consistent {full_leaning}-leaning; uneven support."

    if full_supported and not clean_supported:
        return f"{full_leaning}-leaning; {full_support} support in FULL only."

    if clean_supported and not full_supported:
        return f"{clean_leaning}-leaning; {clean_support} support in CLEAN only."

    return "No clear split in either corpus."


def _pair_note_for_matrix(term1: Any, term2: Any) -> str:
    """
    Return appendix-facing interpretive notes.

    These are not statistical evidence tiers.
    """
    pair_key = _paper_pair_key(term1, term2)

    if pair_key == _paper_pair_key("wabi", "sabi"):
        return "conceptual overlap"

    if pair_key == _paper_pair_key("tokonoma", "chigaidana"):
        return "limited corpus basis"

    return ""

def _matrix_cell_class(text: str) -> str:
    """Map audience-facing table text to visual classes."""
    value = str(text).strip().lower()

    if not value:
        return "missing"

    if value in {"missing", "not tested", "not available"}:
        return "missing"

    if "direction changes" in value or "direction differs" in value or "conflict" in value:
        return "conflict"

    if "conceptual overlap" in value or "near-fusion" in value:
        return "conceptual_overlap"

    if "limited corpus basis" in value or "low-support" in value or "restraint" in value:
        return "limited_basis"

    if "no clear split" in value or "crosses 0" in value or "no clear" in value:
        return "no_clear"

    if "suggestive" in value:
        return "suggestive_support"

    if "strong" in value or "excludes 0" in value:
        return "strong_support"

    return "missing"

    """
    Return a purely statistical FULL/CLEAN status.

    This deliberately does not encode planned paper roles such as near-fusion
    or restraint case. Those are handled separately as paper interpretation.
    """
    full_row = _mode_row_for_pair_concept(df, term1, term2, concept, "FULL")
    clean_row = _mode_row_for_pair_concept(df, term1, term2, concept, "CLEAN")

    full_stable = _row_stable(full_row)
    clean_stable = _row_stable(clean_row)

    full_direction = _row_direction(full_row)
    clean_direction = _row_direction(clean_row)

    if full_stable and clean_stable:
        if full_direction and clean_direction and full_direction != clean_direction:
            return "conflicting_direction", ""
        return "stable_both", full_direction or clean_direction

    if full_stable and not clean_stable:
        return "full_only", full_direction

    if clean_stable and not full_stable:
        return "clean_only", clean_direction

    return "unstable_both", ""


def _paper_role_for_pair(term1: Any, term2: Any) -> str:
    """Return the planned paper role for a fixed Paper 3 pair."""
    key = _paper_pair_key(term1, term2)

    role_lookup = {
        _paper_pair_key("engawa", "veranda"): "cross_cultural_architectural_stress_test",
        _paper_pair_key("tsuboniwa", "impluvium"): "secondary_cross_cultural_architectural_stress_test",
        _paper_pair_key("shōji", "fusuma"): "within_culture_architectural_benchmark",
        _paper_pair_key("tokonoma", "chigaidana"): "restraint_low_visibility_case",
        _paper_pair_key("ma", "mu"): "conceptual_differentiation",
        _paper_pair_key("wabi", "sabi"): "conceptual_near_fusion",
    }

    return role_lookup.get(key, "not_in_fixed_paper_battery")


def _corpus_support_fields(row: pd.Series) -> Dict[str, Any]:
    """
    Return neutral corpus-support fields for the paper CSV.

    These fields describe online documentation density. They should not be read
    as methodological failure or sampling imbalance.
    """
    term1 = str(row.get("term_1", "")).strip()
    term2 = str(row.get("term_2", "")).strip()

    c1 = _num(row.get("word1_contexts"))
    c2 = _num(row.get("word2_contexts"))

    if not np.isfinite(c1) or not np.isfinite(c2):
        return {
            "corpus_support_level": "unavailable",
            "documentation_ratio": float("nan"),
            "lower_visibility_term": "",
            "interpretive_scope_note": "corpus_support_unavailable",
        }

    low = min(c1, c2)
    high = max(c1, c2)

    if low <= 0:
        ratio = float("inf")
    else:
        ratio = float(high / low)

    if c1 < c2:
        lower_term = term1
    elif c2 < c1:
        lower_term = term2
    else:
        lower_term = "equal"

    if low < 25:
        support_level = "limited"
    elif low < 50:
        support_level = "moderate"
    else:
        support_level = "high"

    notes: List[str] = []

    if support_level == "limited":
        notes.append("limited_online_documentation_for_lower_visibility_term")

    if np.isfinite(ratio) and ratio >= 3.0:
        notes.append("asymmetric_online_documentation_density")

    if not notes:
        notes.append("adequate_online_documentation_support")

    return {
        "corpus_support_level": support_level,
        "documentation_ratio": ratio,
        "lower_visibility_term": lower_term,
        "interpretive_scope_note": "; ".join(notes),
    }

def build_paper_master_results_csv(df: pd.DataFrame, master_dir: Path) -> Path:
    """
    Write a paper-only master CSV.

    This is not the diagnostic long-form master CSV. It contains only the fields
    needed for paper figures and interpretation.

    Important:
    - Primary p-values and FDR corrections are included so paper-facing labels
      can be audited against the inferential table.
    - Corpus support is reported as online documentation density, not as a sampling defect.
    """
    master_dir.mkdir(parents=True, exist_ok=True)

    out_path = master_dir / "paper_master_results.csv"

    paper_columns = [
        "comparison_family",
        "pair_type",
        "paper_role",
        "term_1",
        "term_2",
        "mode",
        "concept_axis",
        "concept_axis_raw",
        "observed_difference",
        "ci_95_low",
        "ci_95_high",
        "ci_excludes_zero",
        "primary_p_value_source",
        "p_perm_label",
        "p_value_fdr_bh",
        "p_value_holm",
        "p_value_fdr_by",
        "significant_p_value_fdr_bh_0.05",
        "target_context_sampling",
        "target_sample_n_word1",
        "target_sample_n_word2",
        "statistical_status",
        "statistical_direction",
        "paper_facing_stability_class",
        "paper_facing_direction",
        "directional_reading",
        "word1_contexts",
        "word2_contexts",
        "contexts_concept_total",
        "corpus_support_level",
        "documentation_ratio",
        "lower_visibility_term",
        "interpretive_scope_note",
        "pair_global_defined",
        "pair_concept_defined",
        "pair_concept_mean_cos",
        "pair_concept_nn_mean_sym",
        "paper_interpretation",
    ]

    needed = {"mode", "term_1", "term_2", "concept", "observed_difference", "ci_95_low", "ci_95_high"}

    if df.empty or not needed.issubset(df.columns):
        pd.DataFrame(columns=paper_columns).to_csv(out_path, index=False)
        LOGGER.warning("Paper master results CSV written empty because required columns are missing.")
        return out_path

    fixed_pair_keys = {
        _paper_pair_key(t1, t2)
        for _family, t1, t2, _role in _fixed_paper_pair_specs()
    }

    work = df.copy()
    work["_mode_norm"] = work["mode"].astype(str).str.upper()
    work = work[work["_mode_norm"].isin(["FULL", "CLEAN"])].copy()

    rows: List[Dict[str, Any]] = []

    for _, row in work.iterrows():
        term1 = str(row.get("term_1", "")).strip()
        term2 = str(row.get("term_2", "")).strip()
        concept = str(row.get("concept", "")).strip()
        pair_key = _paper_pair_key(term1, term2)

        if pair_key not in fixed_pair_keys:
            continue

        family, pair_type = _paper_pair_metadata(term1, term2)
        interp = PAPER3_PAIR_INTERPRETATIONS.get(pair_key, pair_type)

        statistical_status, statistical_direction = _statistical_status_for_pair_concept(
            work,
            term1,
            term2,
            concept,
        )

        paper_facing_class, paper_facing_direction = _stability_class_for_pair_concept(
            work,
            term1,
            term2,
            concept,
        )

        lo = _num(row.get("ci_95_low"))
        hi = _num(row.get("ci_95_high"))
        stable = _ci_excludes_zero(lo, hi)

        support_fields = _corpus_support_fields(row)

        rows.append({
            "comparison_family": family,
            "pair_type": pair_type,
            "paper_role": _paper_role_for_pair(term1, term2),
            "term_1": term1,
            "term_2": term2,
            "mode": str(row.get("mode", "")).upper(),
            "concept_axis": _plain_label(concept),
            "concept_axis_raw": concept,
            "observed_difference": _num(row.get("observed_difference")),
            "ci_95_low": lo,
            "ci_95_high": hi,
            "ci_excludes_zero": bool(stable),
            "primary_p_value_source": row.get("primary_p_value_source", ""),
            "p_perm_label": _num(row.get("p_perm_label")),
            "p_value_fdr_bh": _num(row.get("p_value_fdr_bh")),
            "p_value_holm": _num(row.get("p_value_holm")),
            "p_value_fdr_by": _num(row.get("p_value_fdr_by")),
            "significant_p_value_fdr_bh_0.05": row.get("significant_p_value_fdr_bh_0.05", ""),
            "target_context_sampling": row.get("target_context_sampling", ""),
            "target_sample_n_word1": _num(row.get("target_sample_n_word1")),
            "target_sample_n_word2": _num(row.get("target_sample_n_word2")),
            "statistical_status": statistical_status,
            "statistical_direction": statistical_direction,
            "paper_facing_stability_class": paper_facing_class,
            "paper_facing_direction": paper_facing_direction,
            "directional_reading": _paper_directional_reading(row),
            "word1_contexts": _num(row.get("word1_contexts")),
            "word2_contexts": _num(row.get("word2_contexts")),
            "contexts_concept_total": _num(row.get("contexts_concept_total")),
            "corpus_support_level": support_fields["corpus_support_level"],
            "documentation_ratio": support_fields["documentation_ratio"],
            "lower_visibility_term": support_fields["lower_visibility_term"],
            "interpretive_scope_note": support_fields["interpretive_scope_note"],
            "pair_global_defined": row.get("pair_global_defined", ""),
            "pair_concept_defined": row.get("pair_concept_defined", ""),
            "pair_concept_mean_cos": _num(row.get("pair_concept_mean_cos")),
            "pair_concept_nn_mean_sym": _num(row.get("pair_concept_nn_mean_sym")),
            "paper_interpretation": interp,
        })

    paper_df = pd.DataFrame(rows, columns=paper_columns)

    if not paper_df.empty:
        family_order = {"architecture": 0, "aesthetics": 1, "other": 2}
        pair_order = {
            _paper_pair_key(t1, t2): idx
            for idx, (_family, t1, t2, _role) in enumerate(_fixed_paper_pair_specs())
        }
        mode_order = {"FULL": 0, "CLEAN": 1}
        concept_order_lookup = {
            concept: idx
            for idx, concept in enumerate(PAPER3_ARCHITECTURE_AXIS_ORDER + PAPER3_AESTHETIC_AXIS_ORDER)
        }

        paper_df["_family_order"] = paper_df["comparison_family"].map(family_order).fillna(99)
        paper_df["_pair_order"] = [
            pair_order.get(_paper_pair_key(t1, t2), 99)
            for t1, t2 in zip(paper_df["term_1"], paper_df["term_2"])
        ]
        paper_df["_mode_order"] = paper_df["mode"].map(mode_order).fillna(99)
        paper_df["_concept_order"] = paper_df["concept_axis_raw"].map(concept_order_lookup).fillna(99)

        paper_df = (
            paper_df.sort_values(
                by=["_family_order", "_pair_order", "_mode_order", "_concept_order", "concept_axis"]
            )
            .drop(columns=["_family_order", "_pair_order", "_mode_order", "_concept_order"])
            .reset_index(drop=True)
        )

    paper_df.to_csv(out_path, index=False, float_format="%.6f")
    LOGGER.info("Wrote paper-only master results CSV to %s", out_path)
    return out_path


def build_publication_figures_pdf(pub_dir: Path) -> Optional[Path]:
    """Combine only publication figures into one small PDF under the publication figure folder."""
    pub_dir.mkdir(parents=True, exist_ok=True)
    out_pdf = pub_dir / "publication_figures.pdf"

    image_paths = [
        pub_dir / filename
        for filename, _title in PAPER3_PUBLICATION_FIGURES
        if (pub_dir / filename).is_file()
    ]

    if not image_paths:
        return None

    with PdfPages(out_pdf) as pdf:
        for img_path in image_paths:
            if not img_path.is_file():
                LOGGER.warning("Missing publication figure: %s", img_path)
                continue

            if img_path.stat().st_size < 10_000:
                LOGGER.warning("Suspiciously small publication figure: %s", img_path)
                continue

            LOGGER.info("Adding publication figure to publication PDF: %s", img_path)

            try:
                img = mpimg.imread(str(img_path))
            except Exception as exc:
                LOGGER.warning("Could not read publication image %s: %s", img_path, exc)
                continue

            height, width = img.shape[0], img.shape[1]
            figsize = (11.69, 8.27) if width >= height else (8.27, 11.69)

            fig = plt.figure(figsize=figsize)
            ax = fig.add_subplot(111)
            ax.imshow(img)
            ax.axis("off")
            pdf.savefig(fig, dpi=220, bbox_inches="tight")
            plt.close(fig)

    LOGGER.info("Wrote publication figures PDF to %s", out_pdf)
    return out_pdf

def _is_under(path: Path, parent: Path) -> bool:
    """Return True if path is inside parent."""
    try:
        path.resolve().relative_to(parent.resolve())
        return True
    except ValueError:
        return False


def _display_relpath(path: Path, root: Path) -> str:
    """Return a readable path relative to root when possible."""
    try:
        return str(path.relative_to(root))
    except ValueError:
        return str(path)


def build_master_results_csv() -> Path:
    """
    Flatten every Paper 3 CSV under the current run folder into one long-form master CSV.

    Writes:
        results_paper3/<run_tag>/master/master_results_long.csv
    """
    MASTER.mkdir(parents=True, exist_ok=True)

    master_rows: List[Dict[str, Any]] = []

    for csv_path in sorted(RES.rglob("*.csv")):
        if _is_under(csv_path, MASTER):
            continue

        try:
            df = pd.read_csv(csv_path)
        except Exception as exc:
            LOGGER.warning("Could not read %s for master CSV: %s", csv_path, exc)
            continue

        if df.empty and len(df.columns) == 0:
            continue

        work = df.copy()

        if len(work.columns) > 0 and str(work.columns[0]).startswith("Unnamed:"):
            work = work.rename(columns={work.columns[0]: "__index__"})

        id_candidates = [
            "__index__",
            "bootstrap_group",
            "mode",
            "model",
            "term_1",
            "term_2",
            "concept",
            "pair",
        ]
        id_cols = [c for c in id_candidates if c in work.columns]

        if not id_cols and len(work.columns) > 0:
            id_cols = [work.columns[0]]

        value_cols = [c for c in work.columns if c not in id_cols]

        if not value_cols and len(work.columns) == 1:
            value_cols = [work.columns[0]]
            id_cols = []

        for row_number, (row_idx, row) in enumerate(work.iterrows()):
            row_context = {
                c: (None if pd.isna(row[c]) else str(row[c]))
                for c in id_cols
                if c in work.columns
            }
            row_context_json = json.dumps(row_context, ensure_ascii=False)
            source_row_index = "" if pd.isna(row_idx) else str(row_idx)

            if value_cols:
                for metric in value_cols:
                    master_rows.append(
                        {
                            "source_file": _display_relpath(csv_path, RES),
                            "source_stem": csv_path.stem,
                            "row_number": row_number,
                            "source_row_index": source_row_index,
                            "row_context": row_context_json,
                            "metric": str(metric),
                            "value": row[metric],
                        }
                    )
            else:
                master_rows.append(
                    {
                        "source_file": _display_relpath(csv_path, RES),
                        "source_stem": csv_path.stem,
                        "row_number": row_number,
                        "source_row_index": source_row_index,
                        "row_context": row_context_json,
                        "metric": "",
                        "value": "",
                    }
                )

    out_path = MASTER / "master_results_long.csv"
    columns = [
        "source_file",
        "source_stem",
        "row_number",
        "source_row_index",
        "row_context",
        "metric",
        "value",
    ]

    if master_rows:
        pd.DataFrame(master_rows, columns=columns).to_csv(out_path, index=False)
    else:
        pd.DataFrame(columns=columns).to_csv(out_path, index=False)

    LOGGER.info("Wrote master long-form results CSV to %s", out_path)
    return out_path

    """
    Combine every generated Paper 3 PNG figure into one master PDF.

                va="center",
                fontsize=12,
            )
            pdf.savefig(fig, dpi=180, bbox_inches="tight")
            plt.close(fig)
        else:
            for img_path in image_paths:
                try:
                    img = mpimg.imread(str(img_path))

    LOGGER.info("Wrote master plots PDF to %s", pdf_path)
    return pdf_path

def _run_bert_block_for_mode(mode: str,
                             bert_corpus: Path,
                             group_name: str,
                             pairs: List[Tuple[str, str]],
                             concepts: Dict[str, List[str]],
                             all_rows: List[Dict[str, Any]]) -> None:
    """Execute the BERT pipeline for one bootstrap group and one mode using its own corpus."""
    validate_corpus(bert_corpus)

    with Timer(f"[{mode}] Load & preprocess BERT corpus"):
        nlp = spacy.load("en_core_web_sm", disable=["ner", "parser"])
        tokenized_sents, raw_sents, lemma_sents = build_tokenized_and_raw(bert_corpus, nlp)

    if not tokenized_sents:
        LOGGER.error("[%s] Tokenized corpus is empty; check the corpus path and filters.", mode)
        return

    LOGGER.info("[%s] Attempting to load BERT from: %s", mode, cache_root_for(mode) / "bert_cased")
    model_bert, tok_bert, bert_src = load_best_epoch_bert(mode)
    record_bert_checkpoint_audit(mode, bert_src)
    LOGGER.info("[%s] BERT model device: %s", mode, next(model_bert.parameters()).device)
    LOGGER.info("[%s] BERT source: %s", mode, bert_src)

    # Fast precompute path for contexts bootstrap
    precomputed: Optional[Dict[str, np.ndarray]] = None
    if P3.bootstrap_unit == "contexts":
        needed = set()
        for a, b in pairs:
            needed.add(a); needed.add(b)
        for lst in concepts.values():
            for w in lst:
                needed.add(w)
        needed_list = sorted(w.lower() for w in needed)
        corpus_hash = sha256_file(bert_corpus)
        cache_dir = Path(P3.precompute_cache_dir)
        cache_key = _cache_key_for_precompute(needed_list, bert_src, BERT_MAX_CTX, BERT_USE_LAST, corpus_hash)
        LOGGER.info("[%s] BERT cache hash: %s", mode, cache_key)
        cache_path = cache_dir / f"{cache_key}.npz"

        precomputed = _load_precomputed(cache_path)
        if precomputed is None:
            LOGGER.info("[%s] Precomputing contextual embeddings for %d terms (batch=%d, fp16=%s)",
                        mode, len(needed_list), int(P3.precompute_batch_size), bool(P3.bert.fp16_inference))
            precomputed = precompute_bert_occurrence_embeddings(
                needed_list, raw_sents, lemma_sents,
                model_bert, tok_bert,
                BERT_MAX_CTX, BERT_USE_LAST,
                batch_size=int(P3.precompute_batch_size),
                fp16=bool(P3.bert.fp16_inference),
            )
            _save_precomputed(cache_path, precomputed)
            LOGGER.info("[%s] Saved precomputed embeddings -> %s", mode, cache_path)
        else:
            LOGGER.info("[%s] Loaded precomputed embeddings -> %s", mode, cache_path)

    global_nn_done: Set[Tuple[str, str, str, str]] = set()
    context_audit_rows: List[Dict[str, Any]] = []

    # Bootstraps for this group only
    for term1, term2 in pairs:
        for concept_name, concept_words in concepts.items():
            if precomputed is not None:
                obs, lo, hi, diffs = bert_bootstrap_diff_precomputed(
                    term1,
                    term2,
                    concept_words,
                    N_BOOT,
                    precomputed,
                    target_context_sampling=P3.target_context_sampling,
                )

                n_word1 = int(precomputed.get(term1.lower(), np.empty((0, 1))).shape[0])
                n_word2 = int(precomputed.get(term2.lower(), np.empty((0, 1))).shape[0])

                if P3.target_context_sampling == "balanced" and n_word1 > 0 and n_word2 > 0:
                    target_sample_n_word1 = min(n_word1, n_word2)
                    target_sample_n_word2 = min(n_word1, n_word2)
                else:
                    target_sample_n_word1 = n_word1
                    target_sample_n_word2 = n_word2

                cover = {
                    "word1_contexts": n_word1,
                    "word2_contexts": n_word2,
                    "target_sample_n_word1": int(target_sample_n_word1),
                    "target_sample_n_word2": int(target_sample_n_word2),
                    "contexts_concept_total": int(sum(precomputed.get(w.lower(), np.empty((0, 1))).shape[0] for w in concept_words)),
                    "target_context_sampling": P3.target_context_sampling,
                }
            else:
                obs, lo, hi, diffs, counts = bert_bootstrap_diff(
                    term1, term2, raw_sents, lemma_sents, concept_words, N_BOOT,
                    model_bert, tok_bert, BERT_MAX_CTX, BERT_USE_LAST
                )
                cover = {
                    "word1_contexts": counts.get("word1_contexts", 0),
                    "word2_contexts": counts.get("word2_contexts", 0),
                    "target_sample_n_word1": counts.get("word1_contexts", 0),
                    "target_sample_n_word2": counts.get("word2_contexts", 0),
                    "contexts_concept_total": counts.get("concept_contexts", 0),
                    "target_context_sampling": "sentence_resample",
                }

            append_context_audit_rows(
                audit_rows=context_audit_rows,
                raw_sents=raw_sents,
                lemma_sents=lemma_sents,
                mode_name=mode,
                group_name=group_name,
                term1=term1,
                term2=term2,
                concept_name=concept_name,
                concept_words=concept_words,
                max_examples_per_item=6,
            )

            stats = _summary_from_bootstrap(diffs, obs, P3.permutation_iterations)
            cohens_d = float(obs / stats["boot_sd"]) if np.isfinite(stats["boot_sd"]) and stats["boot_sd"] > 0 else 0.0

            extra: Dict[str, Any] = {}

            # === Within-Pair Interchangeability (WPI) ===
            wpi_cfg = _get_interchangeability_cfg(CFG)
            wpi_on = bool(wpi_cfg.get("enabled", True))
            concept_q = float(wpi_cfg.get("concept_quantile", 0.67))
            min_inst = int(wpi_cfg.get("min_instances", 3))
            _raw = wpi_cfg.get("bootstrap_iterations")
            if _raw is None or str(_raw).strip().lower() in {"", "null"}:
                wpi_b = N_BOOT
            else:
                wpi_b = int(_raw)
            wpi_plots = bool(wpi_cfg.get("write_plots", True))

            # Gather per-occurrence matrices for the pair and for the concept pool
            if precomputed is not None:
                w1_mat = precomputed.get(term1.lower(), np.empty((0, 768), dtype=np.float32))
                w2_mat = precomputed.get(term2.lower(), np.empty((0, 768), dtype=np.float32))
                c_pool = [precomputed.get(w.lower(), np.empty((0, 768), dtype=np.float32)) for w in concept_words]
            else:
                w1_mat = collect_occurrence_embeddings_for_word(term1, raw_sents, lemma_sents, model_bert, tok_bert,
                                                                BERT_MAX_CTX, BERT_USE_LAST)
                w2_mat = collect_occurrence_embeddings_for_word(term2, raw_sents, lemma_sents, model_bert, tok_bert,
                                                                BERT_MAX_CTX, BERT_USE_LAST)
                c_pool = [collect_occurrence_embeddings_for_word(w, raw_sents, lemma_sents, model_bert, tok_bert,
                                                                 BERT_MAX_CTX, BERT_USE_LAST) for w in concept_words]

            pair_seed = int(hashlib.md5(f"{SEED}|{mode}|{term1}|{term2}|{concept_name}".encode()).hexdigest(), 16) % (
                        2 ** 32)
            rng_local = np.random.default_rng(pair_seed)

            # Global WPI (no concept filter)
            g_centroid, g_nn, g_asym, g_n1, g_n2 = wpi_point_estimates(w1_mat, w2_mat)
            g_boot = wpi_bootstrap(w1_mat, w2_mat, wpi_b, rng_local, g_asym) if wpi_on else {
                "wpi_nn_ci_low": np.nan, "wpi_nn_ci_high": np.nan,
                "wpi_asym_ci_low": np.nan, "wpi_asym_ci_high": np.nan,
                "wpi_asym_p_boot_2sided": np.nan
            }

            # Concept-conditioned WPI (keep top quantile around the concept centroid)
            concept_centroid = _concept_centroid_from_pool(c_pool)
            p_perm_label = label_permutation_p_precomputed(
                w1_mat=w1_mat,
                w2_mat=w2_mat,
                concept_centroid=concept_centroid,
                observed=float(obs),
                n_perm=int(P3.permutation_iterations),
                seed=pair_seed,
                target_context_sampling=P3.target_context_sampling if precomputed is not None else "observed",
            )
            w1_c = _filter_by_concept_quantile(w1_mat, concept_centroid, concept_q)
            w2_c = _filter_by_concept_quantile(w2_mat, concept_centroid, concept_q)

            concept_filter_changed = (
                    w1_c.shape[0] != w1_mat.shape[0] or
                    w2_c.shape[0] != w2_mat.shape[0]
            )

            c_centroid, c_nn, c_asym, c_n1, c_n2 = wpi_point_estimates(w1_c, w2_c)
            c_boot = wpi_bootstrap(w1_c, w2_c, wpi_b, rng_local, c_asym) if wpi_on else {
                "wpi_nn_ci_low": np.nan, "wpi_nn_ci_high": np.nan,
                "wpi_asym_ci_low": np.nan, "wpi_asym_ci_high": np.nan,
                "wpi_asym_p_boot_2sided": np.nan
            }

            pair_global = pairwise_similarity_point_estimates(w1_mat, w2_mat)
            pair_concept = pairwise_similarity_point_estimates(w1_c, w2_c)

            if wpi_b > 0:
                pair_global_boot = pairwise_similarity_bootstrap(w1_mat, w2_mat, wpi_b, rng_local)
                pair_concept_boot = pairwise_similarity_bootstrap(w1_c, w2_c, wpi_b, rng_local)
            else:
                pair_global_boot = {
                    "pair_mean_cos_ci_low": np.nan, "pair_mean_cos_ci_high": np.nan,
                    "pair_nn_xy_ci_low": np.nan, "pair_nn_xy_ci_high": np.nan,
                    "pair_nn_yx_ci_low": np.nan, "pair_nn_yx_ci_high": np.nan,
                    "pair_nn_sym_ci_low": np.nan, "pair_nn_sym_ci_high": np.nan,
                }
                pair_concept_boot = {
                    "pair_mean_cos_ci_low": np.nan, "pair_mean_cos_ci_high": np.nan,
                    "pair_nn_xy_ci_low": np.nan, "pair_nn_xy_ci_high": np.nan,
                    "pair_nn_yx_ci_low": np.nan, "pair_nn_yx_ci_high": np.nan,
                    "pair_nn_sym_ci_low": np.nan, "pair_nn_sym_ci_high": np.nan,
                }

            extra.update({
                "energy_distance": float(_energy_distance(w1_mat, w2_mat)),
                "mmd_rbf": float(_mmd_rbf(w1_mat, w2_mat)),
                "silhouette_two_sample": float(_silhouette_two_sample(w1_mat, w2_mat)),
                "energy_distance_concept": float(_energy_distance(w1_c, w2_c)),
                "mmd_rbf_concept": float(_mmd_rbf(w1_c, w2_c)),
                "silhouette_two_sample_concept": float(_silhouette_two_sample(w1_c, w2_c)),
            })

            # Optional WPI plots
            global_plot_key = (group_name, mode, term1, term2)

            if wpi_plots and g_n1 >= min_inst and g_n2 >= min_inst and global_plot_key not in global_nn_done:
                sims = _pairwise_cos(w1_mat, w2_mat)
                if sims.size:
                    plot_interchangeability(
                        np.max(sims, axis=1),
                        np.max(sims, axis=0),
                        term1, term2, "GLOBAL", FIG, mode, group_name
                    )
                    global_nn_done.add(global_plot_key)

            if wpi_plots and c_n1 >= min_inst and c_n2 >= min_inst:
                sims_c = _pairwise_cos(w1_c, w2_c)
                if sims_c.size:
                    plot_interchangeability(
                        np.max(sims_c, axis=1),
                        np.max(sims_c, axis=0),
                        term1, term2, f"{concept_name}_CONCEPT", FIG, mode, group_name
                    )

            row = {
                "bootstrap_group": group_name,
                "mode": mode, "model": "BERT", "term_1": term1, "term_2": term2,
                "concept": concept_name, "observed_difference": float(obs),
                "ci_95_low": float(lo), "ci_95_high": float(hi),
                "n_bootstraps": int(len(diffs)), "bootstrap_mode": BOOT_MODE,
            }
            row.update({
                # WPI (global)
                "wpi_global_centroid_cos": g_centroid,
                "wpi_global_nn_overlap": g_nn,
                "wpi_global_asym": g_asym,
                "wpi_global_nn_ci_low": g_boot["wpi_nn_ci_low"],
                "wpi_global_nn_ci_high": g_boot["wpi_nn_ci_high"],
                "wpi_global_asym_ci_low": g_boot["wpi_asym_ci_low"],
                "wpi_global_asym_ci_high": g_boot["wpi_asym_ci_high"],
                "wpi_global_asym_p_boot_2sided": g_boot["wpi_asym_p_boot_2sided"],
                "wpi_global_n_word1": int(g_n1),
                "wpi_global_n_word2": int(g_n2),

                # Direct pairwise similarity (global)
                "pair_global_mean_cos": pair_global["mean_cos"],
                "pair_global_mean_cos_ci_low": pair_global_boot["pair_mean_cos_ci_low"],
                "pair_global_mean_cos_ci_high": pair_global_boot["pair_mean_cos_ci_high"],
                "pair_global_nn_mean_xy": pair_global["nn_mean_xy"],
                "pair_global_nn_mean_yx": pair_global["nn_mean_yx"],
                "pair_global_nn_mean_sym": pair_global["nn_mean_sym"],
                "pair_global_nn_mean_xy_ci_low": pair_global_boot["pair_nn_xy_ci_low"],
                "pair_global_nn_mean_xy_ci_high": pair_global_boot["pair_nn_xy_ci_high"],
                "pair_global_nn_mean_yx_ci_low": pair_global_boot["pair_nn_yx_ci_low"],
                "pair_global_nn_mean_yx_ci_high": pair_global_boot["pair_nn_yx_ci_high"],
                "pair_global_nn_mean_sym_ci_low": pair_global_boot["pair_nn_sym_ci_low"],
                "pair_global_nn_mean_sym_ci_high": pair_global_boot["pair_nn_sym_ci_high"],

                # WPI (concept-conditioned)
                "wpi_concept_centroid_cos": c_centroid,
                "wpi_concept_nn_overlap": c_nn,
                "wpi_concept_asym": c_asym,
                "wpi_concept_nn_ci_low": c_boot["wpi_nn_ci_low"],
                "wpi_concept_nn_ci_high": c_boot["wpi_nn_ci_high"],
                "wpi_concept_asym_ci_low": c_boot["wpi_asym_ci_low"],
                "wpi_concept_asym_ci_high": c_boot["wpi_asym_ci_high"],
                "wpi_concept_asym_p_boot_2sided": c_boot["wpi_asym_p_boot_2sided"],
                "wpi_concept_n_word1": int(c_n1),
                "wpi_concept_n_word2": int(c_n2),
                "concept_filter_changed": bool(concept_filter_changed),

                # Direct pairwise similarity (concept-conditioned)
                "pair_concept_mean_cos": pair_concept["mean_cos"],
                "pair_concept_mean_cos_ci_low": pair_concept_boot["pair_mean_cos_ci_low"],
                "pair_concept_mean_cos_ci_high": pair_concept_boot["pair_mean_cos_ci_high"],
                "pair_concept_nn_mean_xy": pair_concept["nn_mean_xy"],
                "pair_concept_nn_mean_yx": pair_concept["nn_mean_yx"],
                "pair_concept_nn_mean_sym": pair_concept["nn_mean_sym"],
                "pair_concept_nn_mean_xy_ci_low": pair_concept_boot["pair_nn_xy_ci_low"],
                "pair_concept_nn_mean_xy_ci_high": pair_concept_boot["pair_nn_xy_ci_high"],
                "pair_concept_nn_mean_yx_ci_low": pair_concept_boot["pair_nn_yx_ci_low"],
                "pair_concept_nn_mean_yx_ci_high": pair_concept_boot["pair_nn_yx_ci_high"],
                "pair_concept_nn_mean_sym_ci_low": pair_concept_boot["pair_nn_sym_ci_low"],
                "pair_concept_nn_mean_sym_ci_high": pair_concept_boot["pair_nn_sym_ci_high"],
            })
            row.update(stats)
            row["p_perm_label"] = float(p_perm_label)
            row.update(cover)
            row.update(extra)
            row["cohens_d_boot"] = cohens_d
            row["target_context_sampling"] = cover.get("target_context_sampling", "")
            row["target_sample_n_word1"] = cover.get("target_sample_n_word1", np.nan)
            row["target_sample_n_word2"] = cover.get("target_sample_n_word2", np.nan)
            row = sanitise_metric_row(row)
            all_rows.append(row)

            if diffs:
                plot_bootstrap_distribution(diffs, obs, (lo, hi), term1, term2, concept_name, FIG, mode, "BERT")

            diag = {
                "bootstrap_group": group_name,
                "pair": [term1, term2],
                "concept": concept_name,
                "observed": obs, "ci_95": [lo, hi],
                "p_value": float(p_perm_label),
                "sd_boot": stats["boot_sd"],
                "cohens_d": cohens_d,
                "n_boot": len(diffs), "bootstrap_mode": BOOT_MODE,
                "contexts": cover,
                "stats": stats,
            }

            # Optional: fast LOO ablation when precomputed embeddings are available
            loo = {}
            if precomputed is not None and concept_words:
                for w_drop in concept_words:
                    cw = [w for w in concept_words if w != w_drop]
                    if not cw:
                        continue
                    obs_loo, lo_loo, hi_loo, _diffs_loo = bert_bootstrap_diff_precomputed(term1, term2, cw, N_BOOT, precomputed)
                    loo[w_drop] = {
                        "observed_without": float(obs_loo),
                        "delta_vs_full": float(obs_loo - obs)
                    }
            diag["concept_loo"] = loo

            diag["wpi"] = {
                "global": {
                    "centroid_cos": g_centroid, "nn_overlap": g_nn, "asym": g_asym,
                    "ci_nn": [row["wpi_global_nn_ci_low"], row["wpi_global_nn_ci_high"]],
                    "ci_asym": [row["wpi_global_asym_ci_low"], row["wpi_global_asym_ci_high"]],
                    "p_asym_2sided": row["wpi_global_asym_p_boot_2sided"],
                    "n_word1": int(g_n1), "n_word2": int(g_n2)
                },
                "concept": {
                    "centroid_cos": c_centroid, "nn_overlap": c_nn, "asym": c_asym,
                    "ci_nn": [row["wpi_concept_nn_ci_low"], row["wpi_concept_nn_ci_high"]],
                    "ci_asym": [row["wpi_concept_asym_ci_low"], row["wpi_concept_asym_ci_high"]],
                    "p_asym_2sided": row["wpi_concept_asym_p_boot_2sided"],
                    "n_word1": int(c_n1), "n_word2": int(c_n2)
                }
            }

            diag["pairwise_similarity"] = {
                "global": {
                    "mean_cos": pair_global["mean_cos"],
                    "ci_mean_cos": [row["pair_global_mean_cos_ci_low"], row["pair_global_mean_cos_ci_high"]],
                    "nn_mean_xy": pair_global["nn_mean_xy"],
                    "nn_mean_yx": pair_global["nn_mean_yx"],
                    "nn_mean_sym": pair_global["nn_mean_sym"],
                    "ci_nn_mean_xy": [row["pair_global_nn_mean_xy_ci_low"], row["pair_global_nn_mean_xy_ci_high"]],
                    "ci_nn_mean_yx": [row["pair_global_nn_mean_yx_ci_low"], row["pair_global_nn_mean_yx_ci_high"]],
                    "ci_nn_mean_sym": [row["pair_global_nn_mean_sym_ci_low"], row["pair_global_nn_mean_sym_ci_high"]],
                },
                "concept": {
                    "mean_cos": pair_concept["mean_cos"],
                    "ci_mean_cos": [row["pair_concept_mean_cos_ci_low"], row["pair_concept_mean_cos_ci_high"]],
                    "nn_mean_xy": pair_concept["nn_mean_xy"],
                    "nn_mean_yx": pair_concept["nn_mean_yx"],
                    "nn_mean_sym": pair_concept["nn_mean_sym"],
                    "ci_nn_mean_xy": [row["pair_concept_nn_mean_xy_ci_low"], row["pair_concept_nn_mean_xy_ci_high"]],
                    "ci_nn_mean_yx": [row["pair_concept_nn_mean_yx_ci_low"], row["pair_concept_nn_mean_yx_ci_high"]],
                    "ci_nn_mean_sym": [row["pair_concept_nn_mean_sym_ci_low"], row["pair_concept_nn_mean_sym_ci_high"]],
                },
                "distribution_separation": {
                    "global": {
                        "energy_distance": row["energy_distance"],
                        "mmd_rbf": row["mmd_rbf"],
                        "silhouette_two_sample": row["silhouette_two_sample"],
                    },
                    "concept": {
                        "energy_distance": row["energy_distance_concept"],
                        "mmd_rbf": row["mmd_rbf_concept"],
                        "silhouette_two_sample": row["silhouette_two_sample_concept"],
                    }
                }
            }

            save_json(
                RES / f"diagnostics_{_safe_tag(group_name)}_{_safe_tag(mode)}_{_safe_tag(term1)}_vs_{_safe_tag(term2)}_{_safe_tag(concept_name)}.json",
                diag
            )

    if context_audit_rows:
        audit_dir = RES / "context_audits"
        audit_dir.mkdir(parents=True, exist_ok=True)

        audit_path = audit_dir / f"context_audit_{_safe_tag(mode)}_{_safe_tag(group_name)}.csv"

        pd.DataFrame(context_audit_rows).to_csv(
            audit_path,
            index=False,
            encoding="utf-8",
        )

        LOGGER.info("Wrote context audit examples to %s", audit_path)

def _run_w2v_block_for_mode(mode: str,
                            w2v_corpus: Path,
                            group_name: str,
                            pairs: List[Tuple[str, str]],
                            concepts: Dict[str, List[str]],
                            all_rows: List[Dict[str, Any]]) -> None:
    """Execute the W2V pipeline (baseline or retrain) for one bootstrap group and one mode using its own corpus."""
    if W2V_BOOTSTRAP_MODE == "off" or not any(m in MODELS_TO_RUN for m in ("cbow", "skipgram")):
        return

    validate_corpus(w2v_corpus)
    with Timer(f"[{mode}] Load & preprocess W2V corpus"):
        nlp = spacy.load("en_core_web_sm", disable=["ner", "parser"])
        tokenized_sents, _, _ = build_tokenized_and_raw(w2v_corpus, nlp)

    cbow_loaded = None
    sg_loaded = None
    if W2V_BOOTSTRAP_MODE == "baseline" and W2V_AVAILABLE and P3.use_validated_cbow_baseline:
        cbow_loaded = load_validated_w2v(mode, "cbow")
        sg_loaded = load_validated_w2v(mode, "sg")

    for term1, term2 in pairs:
        for concept_name, concept_words in concepts.items():
            # CBOW
            if "cbow" in MODELS_TO_RUN:
                if W2V_BOOTSTRAP_MODE == "retrain":
                    obs, lo, hi, diffs = w2v_bootstrap_diff_retrain(
                        term1, term2, tokenized_sents, concept_words, N_BOOT, sg_flag=0, params=W2V_PARAMS
                    )
                    stats = _summary_from_bootstrap(diffs, obs, P3.permutation_iterations)
                    cohens_d = float(obs / stats["boot_sd"]) if np.isfinite(stats["boot_sd"]) and stats["boot_sd"] > 0 else 0.0
                    row = {
                        "bootstrap_group": group_name,
                        "mode": mode, "model": "W2V CBOW", "term_1": term1, "term_2": term2,
                        "concept": concept_name, "observed_difference": float(obs),
                        "ci_95_low": float(lo), "ci_95_high": float(hi),
                        "n_bootstraps": int(len(diffs)), "bootstrap_mode": BOOT_MODE
                    }
                    row.update(stats); row["cohens_d_boot"] = cohens_d
                    all_rows.append(row)
                    if diffs:
                        plot_bootstrap_distribution(diffs, obs, (lo, hi), term1, term2, concept_name, FIG, mode, "W2V_CBOW")
                elif W2V_BOOTSTRAP_MODE == "baseline":
                    base = w2v_baseline_diff_loaded(cbow_loaded, term1, term2, concept_words)
                    if base is not None:
                        all_rows.append({
                            "bootstrap_group": group_name,
                            "mode": mode, "model": "W2V CBOW (validated baseline)", "term_1": term1, "term_2": term2,
                            "concept": concept_name, "observed_difference": float(base),
                            "ci_95_low": np.nan, "ci_95_high": np.nan,
                            "n_bootstraps": 0, "bootstrap_mode": "baseline",
                            "boot_mean": np.nan, "boot_median": np.nan, "boot_sd": np.nan, "boot_se": np.nan,
                            "boot_min": np.nan, "boot_max": np.nan,
                            "q05": np.nan, "q16": np.nan, "q25": np.nan, "q50": np.nan, "q75": np.nan,
                            "q84": np.nan, "q95": np.nan,
                            "skew": np.nan, "kurtosis_excess": np.nan,
                            "p_boot_2sided": np.nan, "p_boot_greater": np.nan, "p_boot_less": np.nan,
                            "cliffs_delta": np.nan, "bias_boot": np.nan, "p_perm_signflip": np.nan,
                            "cohens_d_boot": np.nan,
                        })

            # Skip-gram
            if "skipgram" in MODELS_TO_RUN:
                if W2V_BOOTSTRAP_MODE == "retrain":
                    obs, lo, hi, diffs = w2v_bootstrap_diff_retrain(
                        term1, term2, tokenized_sents, concept_words, N_BOOT, sg_flag=1, params=W2V_PARAMS
                    )
                    stats = _summary_from_bootstrap(diffs, obs, P3.permutation_iterations)
                    cohens_d = float(obs / stats["boot_sd"]) if np.isfinite(stats["boot_sd"]) and stats["boot_sd"] > 0 else 0.0
                    row = {
                        "bootstrap_group": group_name,
                        "mode": mode, "model": "W2V Skip-gram", "term_1": term1, "term_2": term2,
                        "concept": concept_name, "observed_difference": float(obs),
                        "ci_95_low": float(lo), "ci_95_high": float(hi),
                        "n_bootstraps": int(len(diffs)), "bootstrap_mode": BOOT_MODE
                    }
                    row.update(stats); row["cohens_d_boot"] = cohens_d
                    all_rows.append(row)
                    if diffs:
                        plot_bootstrap_distribution(diffs, obs, (lo, hi), term1, term2, concept_name, FIG, mode, "W2V_Skipgram")
                elif W2V_BOOTSTRAP_MODE == "baseline":
                    base = w2v_baseline_diff_loaded(sg_loaded, term1, term2, concept_words)
                    if base is not None:
                        all_rows.append({
                            "bootstrap_group": group_name,
                            "mode": mode, "model": "W2V Skip-gram (validated baseline)", "term_1": term1, "term_2": term2,
                            "concept": concept_name, "observed_difference": float(base),
                            "ci_95_low": np.nan, "ci_95_high": np.nan,
                            "n_bootstraps": 0, "bootstrap_mode": "baseline",
                            "boot_mean": np.nan, "boot_median": np.nan, "boot_sd": np.nan, "boot_se": np.nan,
                            "boot_min": np.nan, "boot_max": np.nan,
                            "q05": np.nan, "q16": np.nan, "q25": np.nan, "q50": np.nan, "q75": np.nan,
                            "q84": np.nan, "q95": np.nan,
                            "skew": np.nan, "kurtosis_excess": np.nan,
                            "p_boot_2sided": np.nan, "p_boot_greater": np.nan, "p_boot_less": np.nan,
                            "cliffs_delta": np.nan, "bias_boot": np.nan, "p_perm_signflip": np.nan,
                            "cohens_d_boot": np.nan,
                        })

def main() -> None:
    """End-to-end entry point for the Paper 3 pipeline."""
    banner(f"PAPER 3 PIPELINE | MODELS={MODELS_TO_RUN} | BOOTSTRAP={BOOT_MODE.upper()} ({N_BOOT} iters)")

    # ensure outputs/logs go into results_paper3/<run_tag>/...
    rebase_outputs_to_run_tag(RUN_TAG)

    # Optional version checks
    if any(asdict(P3.require_versions).values()):
        check_required_versions(P3.require_versions)

    manifest = write_run_manifest({"note": "Paper 3 BERT-first run", "run_tag": RUN_TAG})
    LOGGER.info("Run manifest at %s", manifest)
    log_runtime_report()
    copy_run_inputs_for_repro(Path(args.config), Path(args.filters))

    if args.dry_run:
        banner("DRY RUN")
        LOGGER.info("Resolved corpora to run: %s", P3.corpora)
        LOGGER.info("Exiting due to --dry_run")
        return

    all_rows: List[Dict[str, Any]] = []

    # Iterate each requested mode, and run each bootstrap group on its own corpora
    for mode in P3.corpora:
        paths = resolve_active_paths_from_yaml(CFG, mode_override=mode)

        for group_name, group in BOOTSTRAP_GROUPS.items():
            if _is_control_group(group_name):
                LOGGER.warning("[%s][%s] Skipping diagnostic/control group", mode, group_name)
                continue

            pairs = group["pairs"]
            concepts = group["concepts"]

            if not pairs:
                LOGGER.warning("[%s][%s] No pairs configured, skipping group", mode, group_name)
                continue
            if not concepts:
                LOGGER.warning("[%s][%s] No concepts configured, skipping group", mode, group_name)
                continue

            # BERT block on its BERT-specific corpus
            if "bert" in MODELS_TO_RUN:
                _run_bert_block_for_mode(
                    mode,
                    Path(paths["bert"]),
                    group_name,
                    pairs,
                    concepts,
                    all_rows
                )

            # W2V block on its W2V-specific corpus
            if any(m in MODELS_TO_RUN for m in ("cbow", "skipgram")):
                _run_w2v_block_for_mode(
                    mode,
                    Path(paths["w2v"]),
                    group_name,
                    pairs,
                    concepts,
                    all_rows
                )

    build_context_audit_master_csv()

    # Save results with multiple-testing corrections
    if all_rows:
        df = pd.DataFrame(all_rows)

        if "bootstrap_group" in df.columns:
            df = df[
                ~df["bootstrap_group"]
                .astype(str)
                .str.lower()
                .str.startswith(CONTROL_GROUP_PREFIXES)
            ].copy()

        # Use label-permutation p-values as the primary null-test p-values.
        # Bootstrap intervals remain the uncertainty summary.
        primary_p_col = "p_perm_label" if "p_perm_label" in df.columns else "p_boot_2sided"

        if primary_p_col in df.columns:
            mask = df[primary_p_col].notna().to_numpy(dtype=bool)
            pvals = df.loc[mask, primary_p_col].astype(float).to_numpy().tolist()

            if pvals:
                df["primary_p_value_source"] = primary_p_col
                df.loc[mask, "p_value_fdr_bh"] = fdr_bh(pvals)
                df.loc[mask, "p_value_holm"] = holm_bonferroni(pvals)
                df.loc[mask, "p_value_fdr_by"] = fdr_by(pvals)

                for col in ("p_value_fdr_bh", "p_value_holm", "p_value_fdr_by"):
                    df[f"significant_{col}_0.05"] = df[col].lt(0.05)

        # Add paper-facing display labels while preserving raw concept keys.
        if "concept" in df.columns:
            df["concept_display"] = df["concept"].astype(str).map(_plain_label)

        # 1) full table with everything
        out_full = RES / "bootstrap_results_with_fdr.csv"
        df.to_csv(out_full, index=False, float_format="%.6f")

        # Cross-mode pivot for quick robustness readout.
        # Keep both the raw concept key and the display label so old/new runs remain matchable.
        try:
            pivot = (df
                     .assign(
                         pair=df["term_1"] + " vs " + df["term_2"],
                         concept_display=df["concept"].astype(str).map(_plain_label),
                     )
                     .pivot_table(index=["pair", "concept", "concept_display"],
                                  columns=["mode"],
                                  values=["observed_difference", "ci_95_low", "ci_95_high", "p_value_fdr_bh"],
                                  aggfunc="first"))
            pivot.to_csv(RES / "bootstrap_cross_mode_pivot.csv")
        except (KeyError, ValueError) as err:
            LOGGER.warning("Cross-mode pivot failed: %s", err)

        # 2) compact table that mirrors the JSON diagnostics, easier to read
        compact_cols = [
            "mode","model","term_1","term_2","concept","concept_display",
            "observed_difference","ci_95_low","ci_95_high",
            "p_boot_2sided","p_perm_label","primary_p_value_source","p_value_fdr_bh","p_value_holm","p_value_fdr_by",
            "significant_p_value_fdr_bh_0.05","significant_p_value_holm_0.05","significant_p_value_fdr_by_0.05",
            "n_bootstraps","bootstrap_mode",
            "word1_contexts", "word2_contexts", "target_sample_n_word1", "target_sample_n_word2",
            "contexts_concept_total", "target_context_sampling",
            "coverage_ok", "wpi_global_defined", "wpi_concept_defined",
            "pair_global_defined", "pair_concept_defined", "concept_filter_changed",
            "silhouette_global_defined", "silhouette_concept_defined",
            "boot_mean","boot_sd","cliffs_delta","p_perm_signflip"
        ]
        compact_cols += [c for c in [
            "wpi_global_centroid_cos", "wpi_global_nn_overlap", "wpi_global_asym",
            "wpi_global_nn_ci_low", "wpi_global_nn_ci_high",
            "wpi_global_asym_ci_low", "wpi_global_asym_ci_high", "wpi_global_asym_p_boot_2sided",
            "wpi_concept_centroid_cos", "wpi_concept_nn_overlap", "wpi_concept_asym",
            "wpi_concept_nn_ci_low", "wpi_concept_nn_ci_high",
            "wpi_concept_asym_ci_low", "wpi_concept_asym_ci_high", "wpi_concept_asym_p_boot_2sided",
            "pair_global_mean_cos", "pair_global_mean_cos_ci_low", "pair_global_mean_cos_ci_high",
            "pair_global_nn_mean_xy", "pair_global_nn_mean_yx", "pair_global_nn_mean_sym",
            "pair_global_nn_mean_xy_ci_low", "pair_global_nn_mean_xy_ci_high",
            "pair_global_nn_mean_yx_ci_low", "pair_global_nn_mean_yx_ci_high",
            "pair_global_nn_mean_sym_ci_low", "pair_global_nn_mean_sym_ci_high",
            "pair_concept_mean_cos", "pair_concept_mean_cos_ci_low", "pair_concept_mean_cos_ci_high",
            "pair_concept_nn_mean_xy", "pair_concept_nn_mean_yx", "pair_concept_nn_mean_sym",
            "pair_concept_nn_mean_xy_ci_low", "pair_concept_nn_mean_xy_ci_high",
            "pair_concept_nn_mean_yx_ci_low", "pair_concept_nn_mean_yx_ci_high",
            "pair_concept_nn_mean_sym_ci_low", "pair_concept_nn_mean_sym_ci_high",
            "energy_distance", "mmd_rbf", "silhouette_two_sample",
            "energy_distance_concept", "mmd_rbf_concept", "silhouette_two_sample_concept"
        ] if c in df.columns]

        compact_cols = [c for c in compact_cols if c in df.columns]
        out_compact = RES / "bootstrap_compact.csv"
        df[compact_cols].to_csv(out_compact, index=False, float_format="%.6f")

        # 3) per-pair CSVs (keeps each comparison together)
        pairs_dir = RES / "pairs"
        pairs_dir.mkdir(parents=True, exist_ok=True)
        for (t1, t2), g in df.groupby(["term_1","term_2"], dropna=False):
            g.to_csv(pairs_dir / f"{_safe_tag(t1)}_vs_{_safe_tag(t2)}.csv", index=False, float_format="%.6f")

        bert_df = pd.DataFrame()

        try:
            bert_df = prepare_plot_df(df[df["model"].astype(str).eq("BERT")].copy())

            if not bert_df.empty:
                if "bootstrap_group" not in bert_df.columns:
                    bert_df["bootstrap_group"] = "default"

                for (group_name, mode_name), gm_df in bert_df.groupby(["bootstrap_group", "mode"], dropna=False):
                    safe_group = _safe_tag(str(group_name))
                    safe_mode = _safe_tag(str(mode_name).lower())
                    group_mode_fig = FIG / safe_group / safe_mode
                    group_mode_fig.mkdir(parents=True, exist_ok=True)
                    write_plot_availability_report(gm_df, group_mode_fig)

                    plot_metric_heatmap(
                        gm_df,
                        "observed_difference",
                        f"BERT concept-alignment difference [{group_name}, {mode_name}]",
                        group_mode_fig / "heatmap_observed_difference.png",
                        centre_zero=True,
                    )

                    if "pair_concept_mean_cos" in gm_df.columns:
                        plot_metric_heatmap(
                            gm_df,
                            "pair_concept_mean_cos",
                            f"BERT concept-conditioned mean cosine [{group_name}, {mode_name}]",
                            group_mode_fig / "heatmap_pair_concept_mean_cos.png",
                            centre_zero=False,
                        )

                    if "pair_concept_nn_mean_sym" in gm_df.columns:
                        plot_metric_heatmap(
                            gm_df,
                            "pair_concept_nn_mean_sym",
                            f"BERT concept-conditioned NN symmetry [{group_name}, {mode_name}]",
                            group_mode_fig / "heatmap_pair_concept_nn_mean_sym.png",
                            centre_zero=False,
                        )

                    if "contexts_concept_total" in gm_df.columns:
                        plot_concept_anchor_coverage(gm_df, group_mode_fig)

                    for (model_name, t1, t2), pair_df in gm_df.groupby(
                        ["model", "term_1", "term_2"], dropna=False
                    ):
                        plot_contrast_forest(
                            pair_df,
                            group_mode_fig,
                            str(mode_name),
                            str(model_name),
                            str(t1),
                            str(t2),
                            str(group_name),
                        )

                        pair_ok = (
                            bool(pair_df["pair_global_defined"].iloc[0])
                            if "pair_global_defined" in pair_df.columns
                            else False
                        )

                        if {"pair_concept_mean_cos", "pair_concept_nn_mean_sym"}.issubset(pair_df.columns):
                            valid_profile_rows = pair_df.dropna(
                                subset=["pair_concept_mean_cos", "pair_concept_nn_mean_sym"]
                            )
                        else:
                            valid_profile_rows = pd.DataFrame()

                        concept_ok = len(valid_profile_rows) >= 2

                        if pair_ok and concept_ok:
                            plot_similarity_profile(
                                pair_df,
                                group_mode_fig,
                                str(mode_name),
                                str(model_name),
                                str(t1),
                                str(t2),
                                str(group_name),
                            )

        except Exception as err:
            LOGGER.warning("Summary plotting failed: %s", err)

        try:
            if not bert_df.empty:
                build_publication_figures(
                    bert_df,
                    FIG / "publication",
                )

                paper_required = [
                    FIG / "publication" / "publication_figure_1_cross_cultural_architecture_effects.png",
                    FIG / "publication" / "publication_figure_2_within_culture_architecture_effects.png",
                    FIG / "publication" / "publication_figure_3_conceptual_comparison_effects.png",
                    FIG / "publication" / "publication_figure_4_result_matrix.png",
                    FIG / "publication" / "publication_figure_5_pair_coverage_audit.png",
                    FIG / "publication" / "appendix_table_1_result_statistics.png",
                    FIG / "publication" / "appendix_table_1_result_statistics.svg",
                    FIG / "publication" / "appendix_table_1_result_statistics_source.csv",
                    FIG / "publication" / "publication_figures.pdf",
                    MASTER / "paper_master_plots.pdf",
                    MASTER / "paper_master_results.csv",
                ]

                missing_paper_outputs = [str(path) for path in paper_required if not path.is_file()]

                if missing_paper_outputs:
                    raise RuntimeError(
                        "Paper-facing publication outputs were not created:\n"
                        + "\n".join(missing_paper_outputs)
                    )

        except Exception:
            LOGGER.exception("Publication plotting failed")
            raise

        final_manifest = write_run_manifest({
            "note": "Fine-tuned BERT targeted comparison run",
            "run_tag": RUN_TAG,
            "bert_checkpoint_audit": BERT_CHECKPOINT_AUDIT,
        })
        LOGGER.info("Updated final run manifest at %s", final_manifest)

        banner("FINAL BOOTSTRAP RESULTS (ENHANCED)")
        head_str = df.head(10).to_string(index=False)
        for line in head_str.splitlines():
            LOGGER.info(line)
        LOGGER.info("Saved: %s", out_full)
        LOGGER.info("Saved: %s", out_compact)
        LOGGER.info("Saved per-pair CSVs under: %s", pairs_dir)
    else:
        banner("NO RESULTS")
        LOGGER.warning("No result rows produced; check config and coverage.")

    # Build master artifacts in one dedicated folder.
    # This does not add computations; it only aggregates files already written by Paper 3.
    build_master_results_csv()
    build_master_plots_pdf()

    banner("PAPER 3 PIPELINE COMPLETE")


if __name__ == "__main__":
    main()
